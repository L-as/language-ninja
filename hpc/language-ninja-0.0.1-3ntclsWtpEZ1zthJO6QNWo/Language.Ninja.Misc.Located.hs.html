  108 -- | This datatype represents a value annotated with a source location.
  109 --
  110 --   @since 0.1.0
  111 data Located t
  112   = MkLocated
  113     { _locatedPos :: {-# UNPACK #-} !Position
  114     , _locatedVal ::                !t
  115     }
  116   deriving (Eq, Show, Generic, Functor, Foldable, Traversable)

  118 -- | Construct a 'Located' value directly.
  119 --
  120 --   @since 0.1.0
  121 {-# INLINE makeLocated #-}
  122 makeLocated :: Position -> t -> Located t
  123 makeLocated = MkLocated

  125 -- | Given @path :: 'Maybe' 'Misc.Path'@ and a @text :: 'Text'@, do the
  126 --   following:
  127 --
  128 --   * Remove all @'\r'@ characters from the @text@.
  129 --   * Split the @text@ into chunks that are guaranteed not to contain newlines
  130 --     or whitespace, and which are annotated with their location.
  131 --
  132 --   @since 0.1.0
  133 tokenize :: Maybe Misc.Path -> Text -> [Located Text]
  134 tokenize mpath = removeWhitespace (mpath, 0, 0)
  135 
  136 -- | Read the file at the given 'Misc.Path' and then run 'tokenize' on the
  137 --   resulting 'Text'.
  138 --
  139 --   @since 0.1.0
  140 tokenizeFile :: (Mock.MonadReadFile m) => Misc.Path -> m [Located Text]
  141 tokenizeFile path = tokenize (Just path) <$> Mock.readFile path
  142 
  143 -- | This function is equivalent to @tokenize Nothing@.
  144 --
  145 --   @since 0.1.0
  146 tokenizeText :: Text -> [Located Text]
  147 tokenizeText = tokenize Nothing

  255 -- | Represents a span of source code.
  256 --
  257 --   @since 0.1.0
  258 data Span
  259   = MkSpan !(Maybe Misc.Path) !Offset !Offset
  260   deriving (Eq, Show, Generic)
  261 
  262 -- | Construct a 'Span' from a given start position to a given end position.
  263 --
  264 --   @since 0.1.0
  265 {-# INLINE makeSpan #-}
  266 makeSpan :: Maybe Misc.Path
  267          -- ^ The file in which this span resides, if any.
  268          -> Offset
  269          -- ^ The start offset.
  270          -> Offset
  271          -- ^ The end offset.
  272          -> Span
  273 makeSpan mpath start end = case compareOffset start end of
  274                              GT -> makeSpan mpath end start
  275                              _  -> MkSpan mpath start end
  276 
  277 -- | A lens into the (nullable) path associated with a 'Span'.
  278 --
  279 --   @since 0.1.0
  280 {-# INLINE spanPath #-}
  281 spanPath :: Lens.Lens' Span (Maybe Misc.Path)
  282 spanPath = let helper (MkSpan p s e) = (p, \x -> MkSpan x s e)
  283            in Lens.lens (helper .> fst) (helper .> snd)
  284 
  285 -- | A lens giving the start and end 'Offset's associated with a 'Span'.
  286 --
  287 --   @since 0.1.0
  288 {-# INLINE spanRange #-}
  289 spanRange :: Lens.Lens' Span (Offset, Offset)
  290 spanRange = let helper (MkSpan p s e) = ((s, e), \(s', e') -> MkSpan p s' e')
  291             in Lens.lens (helper .> fst) (helper .> snd)

  321 -- | Converts to @{file: …, start: …, end: …}@.
  322 --
  323 --   @since 0.1.0
  324 instance Aeson.ToJSON Span where
  325   toJSON (MkSpan file start end)
  326     = [ "file"  .= maybe Aeson.Null Aeson.toJSON file
  327       , "start" .= offsetJ start
  328       , "end"   .= offsetJ end
  329       ] |> Aeson.object
  330     where
  331       offsetJ :: (Line, Column) -> Aeson.Value
  332       offsetJ (line, col) = Aeson.object ["line" .= line, "col" .= col]
  333 
  334 -- | Inverse of the 'Aeson.ToJSON' instance.
  335 --
  336 --   @since 0.1.0
  337 instance Aeson.FromJSON Span where
  338   parseJSON = (Aeson.withObject "Span" $ \o -> do
  339                   file  <- (o .: "file")  >>= pure
  340                   start <- (o .: "start") >>= offsetP
  341                   end   <- (o .: "end")   >>= offsetP
  342                   pure (MkSpan file start end))
  343     where
  344       offsetP :: Aeson.Value -> Aeson.Parser Offset
  345       offsetP = (Aeson.withObject "Offset" $ \o -> do
  346                     line <- (o .: "line") >>= pure
  347                     col  <- (o .: "col")  >>= pure
  348                     pure (line, col))

  422 -- | If two 'Position's are comparable (i.e.: if they are in the same file),
  423 --   this function will return an 'Ordering' giving their relative positions.
  424 --   Otherwise, it will of course return 'Nothing'.
  425 --
  426 --   @since 0.1.0
  427 comparePosition :: Position -> Position -> Maybe Ordering
  428 comparePosition = go
  429   where
  430     go (MkPosition fileX lineX colX) (MkPosition fileY lineY colY)
  431       = compareTriple (fileX, (lineX, colX)) (fileY, (lineY, colY))
  432 
  433     compareTriple :: (Maybe Misc.Path, Offset) -> (Maybe Misc.Path, Offset)
  434                   -> Maybe Ordering
  435     compareTriple (mfileX, offX) (mfileY, offY)
  436       | (mfileX == mfileY) = Just (compareOffset offX offY)
  437       | otherwise          = Nothing
  438

  481 -- | A line/column offset into a file.
  482 --
  483 --   @since 0.1.0
  484 type Offset = (Line, Column)
  485 
  486 -- | Compare two 'Offset's in lexicographic order (i.e.: the 'Column' is
  487 --   ignored unless they are on the same 'Line').
  488 --
  489 --   @since 0.1.0
  490 compareOffset :: Offset -> Offset -> Ordering
  491 compareOffset (lineX, colX) (lineY, colY)
  492   | (lineX < lineY) = LT
  493   | (lineX > lineY) = GT
  494   | otherwise       = compare colX colY
  495

  532 data Chunk
  533   = ChunkText                 !Text
  534   | ChunkSpace {-# UNPACK #-} !Int
  535   | ChunkLine  {-# UNPACK #-} !Int
  536   deriving (Eq, Show)
  537 
  538 --------------------------------------------------------------------------------
  539 
  540 newtype Chunks
  541   = MkChunks { fromChunks :: [Chunk] }
  542   deriving (Eq, Show)
  543 
  544 {-# INLINE chunksNil #-}
  545 chunksNil :: Chunks
  546 chunksNil = MkChunks []
  547 
  548 chunksCons :: Chunk -> Chunks -> Chunks
  549 chunksCons = \chunk (MkChunks list) -> MkChunks (go chunk list)
  550   where
  551     {-# INLINE go #-}
  552     go (ChunkSpace m) (ChunkSpace n : rest) = ChunkSpace (m + n)  : rest
  553     go (ChunkLine  m) (ChunkLine  n : rest) = ChunkLine  (m + n)  : rest
  554     go (ChunkText  a) (ChunkText  b : rest) = ChunkText  (a <> b) : rest
  555     go other          list                  = other               : list
  556 
  557 {-# INLINE chunksAddChar #-}
  558 chunksAddChar :: Char -> Chunks -> Chunks
  559 chunksAddChar '\n'             = chunksCons (ChunkLine 1)
  560 chunksAddChar '\r'             = id
  561 chunksAddChar c    | isSpace c = chunksCons (ChunkSpace 1)
  562 chunksAddChar c                = chunksCons (ChunkText (Text.singleton c))

  566 removeWhitespace :: (Maybe Misc.Path, Line, Column) -> Text -> [Located Text]
  567 removeWhitespace (file, initLine, initCol) =
  568   go .> Data.Maybe.catMaybes .> map makeLoc
  569   where
  570     go :: Text -> [Maybe (Line, Column, Text)]
  571     go text = ST.runST $ do
  572       ref <- ST.newSTRef (initLine, initCol)
  573       Text.foldr chunksAddChar chunksNil text
  574         |> fromChunks
  575         |> mapM (applyChunk ref)
  576 
  577     applyChunk :: STRef s (Line, Column)
  578                -> Chunk -> ST s (Maybe (Line, Column, Text))
  579     applyChunk ref = \case
  580       ChunkLine  n -> do ST.modifySTRef' ref ((+ n) *** const 0)
  581                          pure Nothing
  582       ChunkSpace n -> do ST.modifySTRef' ref (second (+ n))
  583                          pure Nothing
  584       ChunkText  t -> do (line, column) <- ST.readSTRef ref
  585                          ST.modifySTRef' ref (second (+ Text.length t))
  586                          pure (Just (line, column, t))
  587 
  588     {-# INLINE makeLoc #-}
  589     makeLoc :: (Line, Column, Text) -> Located Text
  590     makeLoc (line, col, text) = makeLocated (MkPosition file line col) text
