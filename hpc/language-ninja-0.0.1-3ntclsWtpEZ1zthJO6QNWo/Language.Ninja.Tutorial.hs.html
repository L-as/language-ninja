  133 -- $parsing
  134 --
  135 -- To parse a Ninja file, we use the "Language.Ninja.Parse" module.
  136 -- In the simplest case, this amounts to parsing a file:
  137 --
  138 -- @ast <- (Parser.'Parser.parseFileIO' "./build.ninja") :: IO AST.'AST.Ninja'@
  139 --
  140 -- For more specialized use cases, consult the module documentation.
  141 --
  142 -- FIXME: update once we switch parsers

  145 
  146 -- $ast
  147 --
  148 -- Now that we have parsed the Ninja file, we can take a look at the AST:
  149 --
  150 -- @
  151 -- let look lens = Lens.'Lens.view' lens ast
  152 -- look AST.'AST.ninjaRules'     :: 'HashMap' 'Text' AST.'AST.Rule'
  153 -- look AST.'AST.ninjaSingles'   :: 'HashMap' 'Text' AST.'AST.Build'
  154 -- look AST.'AST.ninjaMultiples' :: 'HashMap' ('HashSet' 'Text') AST.'AST.Build'
  155 -- look AST.'AST.ninjaPhonys'    :: 'HashMap' 'Text' ('HashSet' 'Text')
  156 -- look AST.'AST.ninjaDefaults'  :: 'HashSet' 'Text'
  157 -- look AST.'AST.ninjaPools'     :: 'HashMap' 'Text' 'Int'
  158 -- look AST.'AST.ninjaSpecials'  :: 'HashMap' 'Text' 'Text'
  159 -- @
  160 --
  161 -- [@AST.'AST.ninjaRules'@]
  162 --   This field corresponds to the @rule@ declarations in the parsed Ninja file.
  163 --   Specifically, it is a map from rule names (as 'Text') to @AST.'AST.Rule'@s.
  164 --
  165 -- [@AST.'AST.ninjaSingles'@]
  166 --   This field contains the set of all non-phony @build@ declarations with
  167 --   exactly one output. Specifically, it is a map from the build output name
  168 --   to an @AST.'AST.Build'@.
  169 --
  170 -- [@AST.'AST.ninjaMultiples'@]
  171 --   This field contains the set of all non-phony @build@ declarations with
  172 --   two or more outputs. Specifically, it is a map from the set of outputs to
  173 --   the corresponding @AST.'AST.Build'@.
  174 --
  175 -- [@AST.'AST.ninjaPhonys'@]
  176 --   This field contains the set of all phony @build@ declarations, as a map
  177 --   from the output name to the set of dependencies. If a phony @build@ has
  178 --   multiple outputs, it will naturally be expanded to multiple entries in
  179 --   this hash map.
  180 --
  181 -- [@AST.'AST.ninjaDefaults'@]
  182 --   This field contains the set of all targets referenced in @default@
  183 --   declarations.
  184 --
  185 -- [@AST.'AST.ninjaPools'@]
  186 --   This field contains the set of all pools defined in the Ninja file,
  187 --   represented as a mapping from the pool name to the pool depth.
  188 --
  189 -- [@AST.'AST.ninjaSpecials'@]
  190 --   This field contains the set of all "special" top-level variables defined
  191 --   in the Ninja file, as a mapping from the variable name to the variable
  192 --   value. As it stands, this map will only ever have at most two keys:
  193 --   @ninja_required_version@ and @builddir@. For more on these variables, look
  194 --   at the manual <https://ninja-build.org/manual.html#ref_toplevel here>.
  195 --
  196 -- == @AST.'AST.Rule'@
  197 --
  198 -- A value of type @AST.'AST.Rule'@ is essentially the set of variables that
  199 -- are bound in a @rule@ body, represented as a map from the variable ('Text')
  200 -- to its unevaluated definition (@AST.'AST.Expr'@). The underlying 'HashMap'
  201 -- can be extracted with @AST.'AST.ruleBind'@.
  202 --
  203 -- == @AST.'AST.Build'@
  204 --
  205 -- A value of type @AST.'AST.Build'@ contains four pieces of information:
  206 --
  207 -- [@AST.'AST.buildRule' :: 'Lens'' AST.'AST.Build' 'Text'@]
  208 --   The name of the @rule@ associated with this @build@ declaration.
  209 -- [@AST.'AST.buildDeps' :: 'Lens'' AST.'AST.Build' AST.'AST.Deps'@]
  210 --   The set of dependencies for this @build@ declaration.
  211 -- [@AST.'AST.buildEnv' :: 'Lens'' AST.'AST.Build' (AST.'AST.Env' 'Text' 'Text')@]
  212 --   The set of file-level variables in scope when the @build@ declaration
  213 --   was parsed.
  214 -- [@AST.'AST.buildBind' :: 'Lens'' AST.'AST.Build' ('HashMap' 'Text' 'Text')@]
  215 --   The set of bindings (indented @key = value@ pairs) for this @build@
  216 --   declaration.
  217 --
  218 -- == @AST.'AST.Deps'@
  219 --
  220 -- A value of type @AST.'AST.Deps'@ contains three pieces of information:
  221 --
  222 -- [@AST.'AST.depsNormal' :: 'Lens'' AST.'AST.Deps' ('HashSet' 'Text')@]
  223 --   The set of "normal" (explicit) dependencies for a @build@ declaration.
  224 -- [@AST.'AST.depsImplicit' :: 'Lens'' AST.'AST.Deps' ('HashSet' 'Text')@]
  225 --   The set of "implicit" dependencies for a @build@ declaration.
  226 -- [@AST.'AST.depsOrderOnly' :: 'Lens'' AST.'AST.Deps' ('HashSet' 'Text')@]
  227 --   The set of "order-only" dependencies for a @build@ declaration.
  228 --
  229 -- <https://ninja-build.org/manual.html#ref_dependencies This> section of the
  230 -- Ninja manual describes in detail the differences between explicit, implicit,
  231 -- and order-only dependencies.
  232

  235 -- $compiling
  236 --
  237 -- To compile a Ninja AST (@AST.'AST.Ninja'@) to the Ninja intermediate
  238 -- representation, we use @compile@ from the "Language.Ninja.Compile" module.
  239 -- In the simplest case, this looks like:
  240 --
  241 -- @
  242 -- let handleError :: Either Errors.'Errors.CompileError' a -> IO a
  243 --     handleError = 'either' ('fail' . 'show') 'pure'
  244 -- ir <- handleError (Compile.'Compile.compile' ast)
  245 -- @
  246 --
  247 -- Since @Compile.'Compile.compile'@ returns in any monad with an instance
  248 -- of @'MonadError' Errors.'Errors.CompileError'@, it is quite flexible.
  249 --
  250 -- For simplicity in this case, however, we use
  251 -- @Either Errors.'Errors.CompileError' AST.'AST.Ninja'@
  252 -- and convert to @IO@ by calling 'fail' when it fails.
  253 
  254 --------------------------------------------------------------------------------

  256 -- $ir
  257 --
  258 -- The @language-ninja@ intermediate representation, as defined in the
  259 -- "Language.Ninja.IR" module, is a reduced form of the Ninja AST that handles
  260 -- as much of the static semantics of Ninja as possible.
  261 --
  262 -- The Ninja IR does not have any notion of variables/scoping, does not contain
  263 -- unrestricted hash maps or environments, and has the @rule@ declarations
  264 -- inlined into the @build@ nodes, thus eliminating the "polymorphism"
  265 -- associated with the @$in@ and @$out@ Ninja variables.
  266 --
  267 -- Since we have now compiled the Ninja AST, we can take a look at the IR:
  268 --
  269 -- @
  270 -- let look lens = Lens.'Lens.view' lens ir
  271 -- look IR.'IR.ninjaMeta'     :: IR.'IR.Meta'
  272 -- look IR.'IR.ninjaBuilds'   :: 'HashSet' IR.'IR.Build'
  273 -- look IR.'IR.ninjaPhonys'   :: 'HashMap' IR.'IR.Target' ('HashSet' IR.'IR.Target')
  274 -- look IR.'IR.ninjaDefaults' :: 'HashSet' IR.'IR.Target'
  275 -- look IR.'IR.ninjaPools'    :: 'HashSet' IR.'IR.Pool'
  276 -- @
  277 --
  278 -- These fields correspond to more well-typed versions of their counterparts
  279 -- in the AST. The main differences to note are:
  280 --
  281 -- 1. @AST.'AST.ninjaSpecials'@ is used to compute @IR.'IR.ninjaMeta'@.
  282 -- 2. 'Text' has mostly been replaced with @IR.'IR.Target'@ where relevant.
  283 -- 3. @AST.'AST.ninjaSingles'@ and @AST.'AST.ninjaMultiples'@ have been merged
  284 --    into a single field: @IR.'IR.ninjaBuilds'@.
  285 -- 4. @AST.'AST.ninjaRules'@ is gone; the @rule@s are now inlined into the
  286 --    @build@s that use them.
  287 --
  288 -- == @IR.'IR.Meta'@
  289 --
  290 -- A value of type @IR.'IR.Meta'@ contains two pieces of information:
  291 --
  292 -- [@IR.'IR.metaReqVersion' :: 'Lens'' IR.'IR.Meta' (Maybe 'Version')@]
  293 --   The parsed Ninja version required to build this file.
  294 --   This corresponds to the @ninja_required_version@ top-level variable.
  295 -- [@IR.'IR.metaBuildDir' :: 'Lens'' IR.'IR.Meta' (Maybe Misc.'Misc.Path')@]
  296 --   This corresponds to the @builddir@ top-level variable.
  297 --
  298 -- == @IR.'IR.Build'@
  299 --
  300 -- A value of type @IR.'IR.Build'@ contains three pieces of information:
  301 --
  302 -- [@IR.'IR.buildRule' :: 'Lens'' IR.'IR.Build' IR.'IR.Rule'@]
  303 --   The @rule@ associated with this @build@ declaration.
  304 -- [@IR.'IR.buildOuts' :: 'Lens'' IR.'IR.Build' ('HashSet' IR.'IR.Output')@]
  305 --   The set of outputs for this @build@ declaration.
  306 -- [@IR.'IR.buildDeps' :: 'Lens'' IR.'IR.Build' ('HashSet' IR.'IR.Dependency')@]
  307 --   The set of dependencies for this @build@ declaration.
  308 --
  309 -- == @IR.'IR.Rule'@
  310 --
  311 -- A value of type @IR.'IR.Rule'@ contains a lot of information. For brevity,
  312 -- we will simply list the lens names and types; refer to the module
  313 -- documentation ("Language.Ninja.IR.Rule") for more information.
  314 --
  315 -- @
  316 -- IR.'IR.ruleName'         :: 'Lens'' IR.'IR.Rule' 'Text'
  317 -- IR.'IR.ruleCommand'      :: 'Lens'' IR.'IR.Rule' Misc.'Misc.Command'
  318 -- IR.'IR.ruleDescription'  :: 'Lens'' IR.'IR.Rule' (Maybe 'Text')
  319 -- IR.'IR.rulePool'         :: 'Lens'' IR.'IR.Rule' IR.'IR.PoolName'
  320 -- IR.'IR.ruleDepfile'      :: 'Lens'' IR.'IR.Rule' (Maybe Misc.'Misc.Path')
  321 -- IR.'IR.ruleSpecialDeps'  :: 'Lens'' IR.'IR.Rule' (Maybe IR.'IR.SpecialDeps')
  322 -- IR.'IR.ruleGenerator'    :: 'Lens'' IR.'IR.Rule' Bool
  323 -- IR.'IR.ruleRestat'       :: 'Lens'' IR.'IR.Rule' Bool
  324 -- IR.'IR.ruleResponseFile' :: 'Lens'' IR.'IR.Rule' (Maybe IR.'IR.ResponseFile')
  325 -- @
  326 --
  327 -- == @IR.'IR.Pool'@
  328 --
  329 -- A value of type @IR.'IR.Pool'@ has a name (@IR.'IR.poolName'@) and a
  330 -- depth (@IR.'IR.poolDepth'@). This type is correct-by-construction; it should
  331 -- not be possible to construct a @'IR.Pool'@ that does not correspond to a
  332 -- valid pool definition or reference.

  336 -- $printing
  337 --
  338 -- Currently there is a rudimentary pretty-printer for the AST in the
  339 -- "Language.Ninja.Pretty" module. It simply returns 'Text' such that if
  340 --
  341 -- @
  342 -- let pretty = Pretty.'Pretty.prettyNinja'
  343 -- let parse  = Parser.'Parser.parseTextIO'
  344 -- @
  345 --
  346 -- then @pretty >=> parse >=> pretty >=> parse@ should be the same as @pure@,
  347 -- modulo read-only side effects.
