   64 -- | An interned (hash-consed) text type.
   65 --   This is a newtype over 'Interned.InternedText' from the @intern@ package.
   66 --
   67 --   @since 0.1.0
   68 newtype IText
   69   = MkIText Interned.InternedText
   70   deriving (Eq, IsString, Generic)
   71 
   72 -- | Get the 'Text' corresponding to the given 'IText' value.
   73 --
   74 --   prop> internText (uninternText x) = x
   75 --
   76 --   >>> uninternText ("foobar" :: IText)
   77 --   "foobar"
   78 --
   79 --   @since 0.1.0
   80 {-# INLINE uninternText #-}
   81 uninternText :: IText -> Text
   82 uninternText (MkIText i) = Interned.unintern i
   83 
   84 -- | Intern a 'Text' value, resulting in an 'IText' value.
   85 --
   86 --   prop> uninternText (internText x) = x
   87 --
   88 --   >>> internText ("foobar" :: Text)
   89 --   "foobar"
   90 --
   91 --   @since 0.1.0
   92 {-# INLINE internText #-}
   93 internText :: Text -> IText
   94 internText = Interned.intern .> MkIText
   95 
   96 -- | An 'Lens.Iso'' between 'Text' and 'IText'.
   97 --
   98 --   prop> ((fromString x) ^. itext) = T.pack x
   99 --
  100 --   >>> (("foobar" :: Text) ^. itext) :: IText
  101 --   "foobar"
  102 --
  103 --   >>> (("foobar" :: IText) ^. from itext) :: Text
  104 --   "foobar"
  105 --
  106 --   @since 0.1.0
  107 {-# INLINE itext #-}
  108 itext :: Lens.Iso' Text IText
  109 itext = Lens.iso internText uninternText
  110 
  111 -- | The 'Ord' instance in @intern@ compares hashes rather than values.
  112 --
  113 --   @since 0.1.0
  114 instance Ord IText where
  115   compare itA itB = compare (uninternText itA) (uninternText itB)
  116 
  117 -- | Displays an 'IText' such that 'fromString' is inverse to 'show'.
  118 --
  119 --   @since 0.1.0
  120 instance Show IText where
  121   show (MkIText i) = show i
  122 
  123 -- | Inverse of the 'Show' instance.
  124 --
  125 --   @since 0.1.0
  126 instance Read IText where
  127   readsPrec i = readsPrec i .> map (first (fromString .> MkIText))
  128 
  129 -- | Uses the 'Hashable' instance for 'Text'. Not very efficient.
  130 --
  131 --   FIXME: perhaps switch to hashing the identifier, since this is likely
  132 --   pretty hot code given all the @HashMap Target â€¦@ types all over the place.
  133 --
  134 --   @since 0.1.0
  135 instance Hashable IText where
  136   hashWithSalt n = uninternText .> hashWithSalt n
  137 
  138 -- | Defined by @rnf a = seq a ()@, since 'IText' is a newtype of strict types.
  139 --
  140 --   @since 0.1.0
  141 instance NFData IText where
  142   rnf a = seq a ()
  143 
  144 -- | Converts to JSON string via 'uninternText'.
  145 --
  146 --   @since 0.1.0
  147 instance Aeson.ToJSON IText where
  148   toJSON = uninternText .> Aeson.toJSON
  149 
  150 -- | Inverse of the 'Aeson.ToJSON' instance.
  151 --
  152 --   @since 0.1.0
  153 instance Aeson.FromJSON IText where
  154   parseJSON = Aeson.withText "IText" (internText .> pure)
  155 
  156 -- | Converts to JSON string via 'uninternText'.
  157 --
  158 --   @since 0.1.0
  159 instance Aeson.ToJSONKey IText where
  160   toJSONKey = Aeson.toJSONKeyText uninternText
  161 
  162 -- | Inverse of the 'Aeson.ToJSONKey' instance.
  163 --
  164 --   @since 0.1.0
  165 instance Aeson.FromJSONKey IText where
  166   fromJSONKey = Aeson.mapFromJSONKeyFunction internText Aeson.fromJSONKey
  167 
  168 -- | Uses the 'Text' instance.
  169 --
  170 --   @since 0.1.0
  171 instance (Monad m, SC.Serial m Text) => SC.Serial m IText where
  172   series = SC.series |> fmap internText
  173 
  174 -- | Uses the 'Text' instance.
  175 --
  176 --   @since 0.1.0
  177 instance (Monad m, SC.CoSerial m Text) => SC.CoSerial m IText where
  178   coseries = SC.coseries .> fmap (\f int -> f (uninternText int))
