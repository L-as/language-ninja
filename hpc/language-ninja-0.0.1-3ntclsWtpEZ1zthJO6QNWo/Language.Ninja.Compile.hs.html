   73 -- | Compile an parsed Ninja file into a intermediate representation.
   74 --
   75 --   @since 0.1.0
   76 compile :: forall m ann. (MonadError Errors.CompileError m)
   77         => AST.Ninja ann -> m IR.Ninja
   78 compile ast = result
   79   where
   80     result :: m IR.Ninja
   81     result = do
   82       meta     <- metaM
   83       builds   <- buildsM
   84       phonys   <- phonysM
   85       defaults <- defaultsM
   86       pools    <- poolsM
   87 
   88       IR.makeNinja
   89         |> Lens.set IR.ninjaMeta     meta
   90         |> Lens.set IR.ninjaBuilds   builds
   91         |> Lens.set IR.ninjaPhonys   phonys
   92         |> Lens.set IR.ninjaDefaults defaults
   93         |> Lens.set IR.ninjaPools    pools
   94         |> pure
   95 
   96     metaM :: m IR.Meta
   97     metaM = do
   98       let getSpecial :: Text -> Maybe Text
   99           getSpecial name = HM.lookup name (Lens.view AST.ninjaSpecials ast)
  100 
  101       let parseVersion :: Text -> m Ver.Version
  102           parseVersion = Ver.version
  103                          .> either Errors.throwVersionParseFailure pure
  104 
  105       reqversion <- getSpecial "ninja_required_version"
  106                     |> fmap parseVersion
  107                     |> sequenceA
  108       builddir   <- getSpecial "builddir"
  109                     |> fmap Misc.makePath
  110                     |> pure
  111 
  112       IR.makeMeta
  113         |> Lens.set IR.metaReqVersion reqversion
  114         |> Lens.set IR.metaBuildDir   builddir
  115         |> pure
  116 
  117     buildsM :: m (HashSet IR.Build)
  118     buildsM = (multiplesAST <> onHM (first HS.singleton) singlesAST)
  119               |> HM.toList |> mapM compileBuild |> fmap HS.fromList
  120 
  121     phonysM :: m (HashMap IR.Target (HashSet IR.Target))
  122     phonysM = HM.toList phonysAST |> mapM compilePhony |> fmap HM.fromList
  123 
  124     defaultsM :: m (HashSet IR.Target)
  125     defaultsM = HS.toList defaultsAST |> mapM compileDefault |> fmap HS.fromList
  126 
  127     poolsM :: m (HashSet IR.Pool)
  128     poolsM = HM.toList poolsAST |> mapM compilePool |> fmap HS.fromList
  129 
  130     compileBuild :: (HashSet Text, AST.Build ann) -> m IR.Build
  131     compileBuild (outputs, buildAST) = do
  132       let depsAST       = Lens.view AST.buildDeps buildAST
  133       let normalDeps    = HS.toList (Lens.view AST.depsNormal    depsAST)
  134       let implicitDeps  = HS.toList (Lens.view AST.depsImplicit  depsAST)
  135       let orderOnlyDeps = HS.toList (Lens.view AST.depsOrderOnly depsAST)
  136 
  137       outs <- HS.toList outputs |> mapM compileOutput |> fmap HS.fromList
  138       rule <- compileRule (outs, buildAST)
  139       deps <- let compileDep ty dep = compileDependency (dep, ty)
  140               in (\n i o -> HS.fromList (n <> i <> o))
  141                  <$> mapM (compileDep IR.NormalDependency)    normalDeps
  142                  <*> mapM (compileDep IR.NormalDependency)    implicitDeps
  143                  <*> mapM (compileDep IR.OrderOnlyDependency) orderOnlyDeps
  144 
  145       IR.makeBuild rule
  146         |> Lens.set IR.buildOuts outs
  147         |> Lens.set IR.buildDeps deps
  148         |> pure
  149 
  150     compilePhony :: (Text, HashSet Text)
  151                   -> m (IR.Target, HashSet IR.Target)
  152     compilePhony (name, deps) = do
  153       ename <- compileTarget name
  154       edeps <- HS.fromList <$> mapM compileTarget (HS.toList deps)
  155       pure (ename, edeps)
  156 
  157     compileDefault :: Text -> m IR.Target
  158     compileDefault = compileTarget
  159 
  160     compilePool :: (Text, Int) -> m IR.Pool
  161     compilePool pair = case pair of
  162       ("console", 1) -> pure IR.makePoolConsole
  163       ("console", d) -> Errors.throwInvalidPoolDepth d
  164       ("",        _) -> Errors.throwEmptyPoolName
  165       (name,      d) -> do dp <- Misc.makePositive d
  166                                  |> maybe (Errors.throwInvalidPoolDepth d) pure
  167                            pure (IR.makePoolCustom name dp)
  168 
  169     compileRule :: (HashSet IR.Output, AST.Build ann) -> m IR.Rule
  170     compileRule (outputs, buildAST) = do
  171       (name, ruleAST) <- lookupRule buildAST
  172 
  173       let orLookupError :: Text -> Maybe a -> m a
  174           orLookupError var = maybe (Errors.throwRuleLookupFailure var) pure
  175 
  176       let env = computeRuleEnv (outputs, buildAST) ruleAST
  177 
  178       let lookupBind :: Text -> m (Maybe Text)
  179           lookupBind = AST.askEnv env .> pure
  180 
  181       let lookupBind_ :: Text -> m Text
  182           lookupBind_ var = lookupBind var >>= orLookupError var
  183 
  184       command      <- lookupBind_ "command" >>= compileCommand
  185       description  <- lookupBind "description"
  186       pool         <- let buildBind = Lens.view AST.buildBind buildAST
  187                       in (HM.lookup "pool" buildBind <|> AST.askEnv env "pool")
  188                          |> fromMaybe ""
  189                          |> IR.parsePoolName
  190                          |> pure
  191       depfile      <- lookupBind "depfile"
  192                       |> fmap (fmap Misc.makePath)
  193       specialDeps  <- let lookupPrefix = lookupBind "msvc_deps_prefix"
  194                       in ((,) <$> lookupBind "deps" <*> lookupPrefix)
  195                          >>= compileSpecialDeps
  196       generator    <- isJust <$> lookupBind "generator"
  197       restat       <- isJust <$> lookupBind "restat"
  198       responseFile <- do ma <- lookupBind "rspfile"
  199                          mb <- lookupBind "rspfile_content"
  200                          pure ((,) <$> ma <*> mb)
  201                            >>= fmap compileResponseFile .> sequenceA
  202 
  203       IR.makeRule name command
  204         |> Lens.set IR.ruleDescription  description
  205         |> Lens.set IR.rulePool         pool
  206         |> Lens.set IR.ruleDepfile      depfile
  207         |> Lens.set IR.ruleSpecialDeps  specialDeps
  208         |> Lens.set IR.ruleGenerator    generator
  209         |> Lens.set IR.ruleRestat       restat
  210         |> Lens.set IR.ruleResponseFile responseFile
  211         |> pure
  212 
  213     compileSpecialDeps :: (Maybe Text, Maybe Text) -> m (Maybe IR.SpecialDeps)
  214     compileSpecialDeps = (\case (Nothing,     _) -> pure Nothing
  215                                 (Just "gcc",  m) -> goGCC  m
  216                                 (Just "msvc", m) -> goMSVC m
  217                                 (Just d,      _) -> Errors.throwUnknownDeps d)
  218       where
  219         goGCC, goMSVC :: Maybe Text -> m (Maybe IR.SpecialDeps)
  220         goGCC  (Just _) = Errors.throwUnexpectedMSVCPrefix "gcc"
  221         goGCC  Nothing  = pure (Just IR.makeSpecialDepsGCC)
  222         goMSVC (Just m) = pure (Just (IR.makeSpecialDepsMSVC m))
  223         goMSVC Nothing  = pure (Just (IR.makeSpecialDepsMSVC defaultPrefix))
  224 
  225         defaultPrefix = "Note: including file: " :: Text
  226 
  227     compileResponseFile :: (Text, Text) -> m IR.ResponseFile
  228     compileResponseFile (file, content) = do
  229       let path = Misc.makePath file
  230       pure (IR.makeResponseFile path content)
  231 
  232     compileTarget :: Text -> m IR.Target
  233     compileTarget = IR.makeTarget .> pure
  234 
  235     compileOutput :: Text -> m IR.Output
  236     compileOutput name = do
  237       target <- compileTarget name
  238       pure (IR.makeOutput target)
  239 
  240     compileDependency :: (Text, IR.DependencyType) -> m IR.Dependency
  241     compileDependency (name, ty) = do
  242       target <- compileTarget name
  243       pure (IR.makeDependency target ty)
  244 
  245     compileCommand :: Text -> m Misc.Command
  246     compileCommand = Misc.makeCommand .> pure
  247 
  248     lookupRule :: AST.Build ann -> m (Text, AST.Rule ann)
  249     lookupRule buildAST = do
  250       let name = Lens.view AST.buildRule buildAST
  251       ruleAST <- HM.lookup name rulesAST
  252                  |> maybe (Errors.throwBuildRuleNotFound name) pure
  253       pure (name, ruleAST)
  254 
  255     computeRuleEnv :: (HashSet IR.Output, AST.Build ann)
  256                    -> AST.Rule ann
  257                    -> AST.Env Text Text
  258     computeRuleEnv (outs, buildAST) ruleAST = do
  259       let depsAST = Lens.view AST.buildDeps buildAST
  260 
  261       -- FIXME: properly handle implicit/explicit outputs here
  262 
  263       let isExplicitOut :: IR.Output -> Bool
  264           isExplicitOut _ = True
  265 
  266       let explicitOuts = HS.toList outs
  267                          |> filter isExplicitOut
  268                          |> map (Lens.view (IR.outputTarget . IR.targetText))
  269 
  270       let explicitDeps = [ Lens.view AST.depsNormal    depsAST
  271                          , Lens.view AST.depsOrderOnly depsAST
  272                          ] |> mconcat |> HS.toList
  273 
  274       let composeList :: [a -> a] -> (a -> a)
  275           composeList = map Endo .> mconcat .> appEndo
  276 
  277       let quote :: Text -> Text
  278           quote x | Text.any isSpace x = mconcat ["\"", x, "\""]
  279           quote x                      = x
  280 
  281       let bindings = Lens.view AST.buildBind buildAST
  282 
  283       -- the order of adding new environment variables matters
  284       AST.scopeEnv (Lens.view AST.buildEnv buildAST)
  285         |> AST.addEnv "out"        (Text.unwords (map quote explicitOuts))
  286         |> AST.addEnv "in"         (Text.unwords (map quote explicitDeps))
  287         |> AST.addEnv "in_newline" (Text.unlines explicitDeps)
  288         |> composeList (map (uncurry AST.addEnv) (HM.toList bindings))
  289         |> AST.addBinds (HM.toList (Lens.view AST.ruleBind ruleAST))
  290 
  291     rulesAST     :: HashMap Text (AST.Rule ann)
  292     singlesAST   :: HashMap Text (AST.Build ann)
  293     multiplesAST :: HashMap (HashSet Text) (AST.Build ann)
  294     phonysAST    :: HashMap Text (HashSet Text)
  295     defaultsAST  :: HashSet Text
  296     poolsAST     :: HashMap Text Int
  297     rulesAST     = Lens.view AST.ninjaRules     ast
  298     singlesAST   = Lens.view AST.ninjaSingles   ast
  299     multiplesAST = Lens.view AST.ninjaMultiples ast
  300     phonysAST    = Lens.view AST.ninjaPhonys    ast
  301     defaultsAST  = Lens.view AST.ninjaDefaults  ast
  302     poolsAST     = Lens.view AST.ninjaPools     ast
  303 
  304     onHM :: (Eq k', Hashable k')
  305          => ((k, v) -> (k', v')) -> HashMap k v -> HashMap k' v'
  306     onHM f = HM.toList .> map f .> HM.fromList
