   84 -- | The type of errors encountered during compilation.
   85 --
   86 --   @since 0.1.0
   87 data CompileError
   88   = -- | Generic catch-all error constructor. Avoid using this.
   89     --
   90     --   @since 0.1.0
   91     GenericCompileError !Text
   92   | -- | Errors encountered while compiling a 'Meta'.
   93     --
   94     --   @since 0.1.0
   95     CompileMetaError    !CompileMetaError
   96   | -- | Errors encountered while compiling a 'Build'.
   97     --
   98     --   @since 0.1.0
   99     CompileBuildError   !CompileBuildError
  100   | -- | Errors encountered while compiling a 'Rule'.
  101     --
  102     --   @since 0.1.0
  103     CompileRuleError    !CompileRuleError
  104   | -- | Errors encountered while compiling the phony 'HashMap'.
  105     --
  106     --   @since 0.1.0
  107     CompilePhonyError   !CompilePhonyError
  108   | -- | Errors encountered while compiling the default target 'HashSet'.
  109     --
  110     --   @since 0.1.0
  111     CompileDefaultError !CompileDefaultError
  112   | -- | Errors encountered while compiling a 'Pool'.
  113     --
  114     --   @since 0.1.0
  115     CompilePoolError    !CompilePoolError
  116   deriving (Eq, Show, Generic)
  117 
  118 -- | Default instance.
  119 --
  120 --   @since 0.1.0
  121 instance Exception CompileError
  122 
  123 -- | Throw a 'CompileError'.
  124 --
  125 --   @since 0.1.0
  126 throwCompileError :: (MonadError CompileError m) => CompileError -> m a
  127 throwCompileError = throwError
  128 
  129 -- | Throw a generic catch-all 'CompileError'.
  130 --
  131 --   @since 0.1.0
  132 throwGenericCompileError :: (MonadError CompileError m) => Text -> m a
  133 throwGenericCompileError msg = throwCompileError (GenericCompileError msg)
