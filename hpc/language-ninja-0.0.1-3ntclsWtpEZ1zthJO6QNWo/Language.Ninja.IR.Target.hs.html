   74 -- | This type represents a Ninja target name.
   75 --
   76 --   @since 0.1.0
   77 newtype Target
   78   = MkTarget
   79     { _targetIText :: Misc.IText
   80     }
   81   deriving ( Eq, Ord, Show, Read, IsString, Generic, Hashable, NFData
   82            , Aeson.ToJSON, Aeson.FromJSON, Aeson.ToJSONKey, Aeson.FromJSONKey )
   83 
   84 -- | Construct a 'Target' from some 'Text'.
   85 --
   86 --   @since 0.1.0
   87 {-# INLINE makeTarget #-}
   88 makeTarget :: Text -> Target
   89 makeTarget = Lens.view Misc.itext .> MkTarget
   90 
   91 -- | An isomorphism between a 'Target' and its underlying 'IText'.
   92 --
   93 --   @since 0.1.0
   94 {-# INLINE targetIText #-}
   95 targetIText :: Lens.Iso' Target Misc.IText
   96 targetIText = Lens.iso _targetIText MkTarget
   97 
   98 -- | An isomorphism that gives access to a 'Text'-typed view of a 'Target',
   99 --   even though the underlying data has type 'IText'.
  100 --
  101 --   This is equivalent to @targetIText . from itext@.
  102 --
  103 --   @since 0.1.0
  104 {-# INLINE targetText #-}
  105 targetText :: Lens.Iso' Target Text
  106 targetText = targetIText . Lens.from Misc.itext
  107 
  108 -- | Uses the underlying 'IText' instance.
  109 --
  110 --   @since 0.1.0
  111 instance (Monad m, SC.Serial m Text) => SC.Serial m Target where
  112   series = SC.newtypeCons MkTarget
  113 
  114 -- | Uses the underlying 'IText' instance.
  115 --
  116 --   @since 0.1.0
  117 instance (Monad m, SC.CoSerial m Text) => SC.CoSerial m Target where
  118   coseries rs = SC.newtypeAlts rs
  119                 >>- \f -> pure (_targetIText .> f)

  123 -- | A Ninja build output.
  124 --
  125 --   More information is available
  126 --   <https://ninja-build.org/manual.html#ref_outputs here>.
  127 --
  128 --   @since 0.1.0
  129 newtype Output
  130   = MkOutput
  131     { _outputTarget :: Target
  132     }
  133   deriving (Eq, Ord, Show, Read, Generic)
  134 
  135 -- | Construct an 'Output'.
  136 --
  137 --   @since 0.1.0
  138 {-# INLINE makeOutput #-}
  139 makeOutput :: Target
  140            -- ^ The underlying target.
  141            -> Output
  142 makeOutput = MkOutput
  143 
  144 -- | A lens for the 'Target' of an 'Output'.
  145 --
  146 --   @since 0.1.0
  147 {-# INLINE outputTarget #-}
  148 outputTarget :: Lens.Lens' Output Target
  149 outputTarget = Lens.lens _outputTarget
  150                $ \(MkOutput {..}) new -> MkOutput { _outputTarget = new, .. }
  151 
  152 -- | Converts to @{target: …, type: …}@.
  153 --
  154 --   @since 0.1.0
  155 instance Aeson.ToJSON Output where
  156   toJSON (MkOutput {..})
  157     = [ "target" .= _outputTarget
  158       ] |> Aeson.object
  159 
  160 -- | Inverse of the 'Aeson.ToJSON' instance.
  161 --
  162 --   @since 0.1.0
  163 instance Aeson.FromJSON Output where
  164   parseJSON = (Aeson.withObject "Output" $ \o -> do
  165                   _outputTarget <- (o .: "target") >>= pure
  166                   pure (MkOutput {..}))
  167 
  168 -- | Default 'Hashable' instance via 'Generic'.
  169 --
  170 --   @since 0.1.0
  171 instance Hashable Output
  172 
  173 -- | Default 'NFData' instance via 'Generic'.

  175 --   @since 0.1.0
  176 instance NFData Output
  177 
  178 -- | Default 'SC.Serial' instance via 'Generic'.
  179 --
  180 --   @since 0.1.0
  181 instance (Monad m, SC.Serial m Text) => SC.Serial m Output
  182 
  183 -- | Default 'SC.CoSerial' instance via 'Generic'.
  184 --
  185 --   @since 0.1.0
  186 instance (Monad m, SC.CoSerial m Text) => SC.CoSerial m Output

  190 -- | A build dependency.
  191 --
  192 --   More information is available
  193 --   <https://ninja-build.org/manual.html#ref_dependencies here>.
  194 --
  195 --   @since 0.1.0
  196 data Dependency
  197   = MkDependency
  198     { _dependencyTarget :: !Target
  199     , _dependencyType   :: !DependencyType
  200     }
  201   deriving (Eq, Ord, Show, Read, Generic)
  202 
  203 -- | Construct a 'Dependency'.
  204 --
  205 --   @since 0.1.0
  206 {-# INLINE makeDependency #-}
  207 makeDependency :: Target
  208                -- ^ The underlying target.
  209                -> DependencyType
  210                -- ^ The dependency type (normal, implicit, or order-only).
  211                -> Dependency
  212 makeDependency = MkDependency
  213 
  214 -- | A lens for the 'Target' of a 'Dependency'.
  215 --
  216 --   @since 0.1.0
  217 {-# INLINE dependencyTarget #-}
  218 dependencyTarget :: Lens.Lens' Dependency Target
  219 dependencyTarget
  220   = Lens.lens _dependencyTarget
  221     $ \(MkDependency {..}) new -> MkDependency { _dependencyTarget = new, .. }
  222 
  223 -- | A lens for the 'DependencyType' of a 'Dependency'.
  224 --
  225 --   @since 0.1.0
  226 {-# INLINE dependencyType #-}
  227 dependencyType :: Lens.Lens' Dependency DependencyType
  228 dependencyType
  229   = Lens.lens _dependencyType
  230     $ \(MkDependency {..}) new -> MkDependency { _dependencyType = new, .. }
  231 
  232 -- | Converts to @{target: …, type: …}@.
  233 --
  234 --   @since 0.1.0
  235 instance Aeson.ToJSON Dependency where
  236   toJSON (MkDependency {..})
  237     = [ "target" .= _dependencyTarget
  238       , "type"   .= _dependencyType
  239       ] |> Aeson.object
  240 
  241 -- | Inverse of the 'Aeson.ToJSON' instance.
  242 --
  243 --   @since 0.1.0
  244 instance Aeson.FromJSON Dependency where
  245   parseJSON = (Aeson.withObject "Dependency" $ \o -> do
  246                   _dependencyTarget <- (o .: "target") >>= pure
  247                   _dependencyType   <- (o .: "type")   >>= pure
  248                   pure (MkDependency {..}))

  272 -- | The type of a 'Dependency': normal, implicit, or order-only.
  273 --
  274 --   @since 0.1.0
  275 data DependencyType
  276   = -- | A normal dependency. These are listed in the @$in@ variable and changes
  277     --   in the relevant target result in a rule execution.
  278     --
  279     --   @since 0.1.0
  280     NormalDependency
  281   | -- | An order-only dependency. These are listed in the @$in@ variable, but
  282     --   are only rebuilt if there is at least one non-order-only dependency
  283     --   that is out of date.
  284     --
  285     --   FIXME: double check this interpretation of the Ninja manual
  286     --
  287     --   @since 0.1.0
  288     OrderOnlyDependency
  289   deriving (Eq, Ord, Show, Read, Generic)
  290 
  291 -- | A prism for the 'NormalDependency' constructor.
  292 --
  293 --   @since 0.1.0
  294 {-# INLINE _NormalDependency #-}
  295 _NormalDependency :: Lens.Prism' DependencyType ()
  296 _NormalDependency = Lens.prism' (const NormalDependency)
  297                     $ \case NormalDependency -> Just ()
  298                             _                -> Nothing
  299 
  300 -- | A prism for the 'OrderOnlyDependency' constructor.
  301 --
  302 --   @since 0.1.0
  303 {-# INLINE _OrderOnlyDependency #-}
  304 _OrderOnlyDependency :: Lens.Prism' DependencyType ()
  305 _OrderOnlyDependency = Lens.prism' (const OrderOnlyDependency)
  306                        $ \case OrderOnlyDependency -> Just ()
  307                                _                   -> Nothing
  308 
  309 -- | Converts to @"normal"@, @"implicit"@, and @"order-only"@ respectively.
  310 --
  311 --   @since 0.1.0
  312 instance Aeson.ToJSON DependencyType where
  313   toJSON NormalDependency    = "normal"
  314   toJSON OrderOnlyDependency = "order-only"
  315 
  316 -- | Inverse of the 'Aeson.ToJSON' instance.
  317 --
  318 --   @since 0.1.0
  319 instance Aeson.FromJSON DependencyType where
  320   parseJSON = (Aeson.withText "DependencyType" $ \case
  321                   "normal"     -> pure NormalDependency
  322                   "order-only" -> pure OrderOnlyDependency
  323                   owise        -> [ "Invalid dependency type "
  324                                   , "\"", owise, "\"; should be one of "
  325                                   , "[\"normal\", \"order-only\"]"
  326                                   ] |> mconcat |> Text.unpack |> fail)
  327

  331 instance Hashable DependencyType

  346 instance (Monad m) => SC.CoSerial m DependencyType
