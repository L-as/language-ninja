   78 -- | A parsed Ninja @rule@ declaration.
   79 --
   80 --   @since 0.1.0
   81 data Rule ann
   82   = MkRule
   83     { _ruleAnn  :: !ann
   84     , _ruleBind :: !(HashMap Text (AST.Expr ann))
   85     }
   86   deriving (Eq, Show, Generic, Functor, Foldable, Traversable)
   87 
   88 -- | Construct a 'Rule' with all default values
   89 --
   90 --   @since 0.1.0
   91 {-# INLINE makeRule #-}
   92 makeRule :: (Monoid ann) => Rule ann
   93 makeRule = MkRule
   94            { _ruleAnn  = mempty
   95            , _ruleBind = mempty
   96            }
   97 
   98 -- | The set of bindings in scope during the execution of this rule.
   99 --
  100 --   @since 0.1.0
  101 {-# INLINE ruleBind #-}
  102 ruleBind :: Lens.Lens' (Rule ann) (HashMap Text (AST.Expr ann))
  103 ruleBind = Lens.lens _ruleBind
  104            $ \(MkRule {..}) x -> MkRule { _ruleBind = x, .. }
  105 
  106 -- | The usual definition for 'Misc.Annotated'.
  107 --
  108 --   @since 0.1.0
  109 instance Misc.Annotated Rule where
  110   annotation' f = Lens.lens _ruleAnn
  111                   $ \(MkRule {..}) x ->
  112                       MkRule { _ruleAnn = x
  113                              , _ruleBind = HM.map (fmap f) _ruleBind
  114                              , .. }
  115 
  116 -- | Converts to @{ann: …, bind: …}@.
  117 --
  118 --   @since 0.1.0
  119 instance (Aeson.ToJSON ann) => Aeson.ToJSON (Rule ann) where
  120   toJSON (MkRule {..})
  121     = [ "ann"  .= _ruleAnn
  122       , "bind" .= _ruleBind
  123       ] |> Aeson.object
  124 
  125 -- | Inverse of the 'Aeson.ToJSON' instance.
  126 --
  127 --   @since 0.1.0
  128 instance (Aeson.FromJSON ann) => Aeson.FromJSON (Rule ann) where
  129   parseJSON = (Aeson.withObject "Rule" $ \o -> do
  130                   _ruleAnn  <- (o .: "ann")  >>= pure
  131                   _ruleBind <- (o .: "bind") >>= pure
  132                   pure (MkRule {..}))
  133 
  134 -- | Reasonable 'QC.Arbitrary' instance for 'Rule'.
  135 --
  136 --   @since 0.1.0
  137 instance (QC.Arbitrary ann) => QC.Arbitrary (Rule ann) where
  138   arbitrary = MkRule <$> QC.arbitrary <*> QC.arbitrary

  162 -- | The set of constraints required for a given constraint to be automatically
  163 --   computed for a 'Rule'.
  164 --
  165 --   @since 0.1.0
  166 type RuleConstraint (c :: * -> Constraint) (ann :: *)
  167   = ( c (HashMap Text (AST.Expr ann))
  168     , c ann
  169     )
