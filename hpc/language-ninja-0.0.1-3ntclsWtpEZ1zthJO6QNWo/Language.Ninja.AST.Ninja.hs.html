   92 -- | A parsed Ninja file.
   93 --
   94 --   @since 0.1.0
   95 data Ninja ann
   96   = MkNinja
   97     { _ninjaAnn       :: !ann
   98     , _ninjaRules     :: !(HashMap Text (AST.Rule ann))
   99     , _ninjaSingles   :: !(HashMap Output  (AST.Build ann))
  100     , _ninjaMultiples :: !(HashMap Outputs (AST.Build ann))
  101     , _ninjaPhonys    :: !(HashMap Text (HashSet Text))
  102     , _ninjaDefaults  :: !(HashSet Text)
  103     , _ninjaPools     :: !(HashMap Text Int)
  104     , _ninjaSpecials  :: !(HashMap Text Text)
  105     }
  106   deriving (Eq, Show, Generic, Functor, Foldable, Traversable)
  107 
  108 -- | Construct a 'Ninja' with all default values
  109 --
  110 --   @since 0.1.0
  111 {-# INLINE makeNinja #-}
  112 makeNinja :: (Monoid ann) => Ninja ann
  113 makeNinja = MkNinja
  114             { _ninjaAnn       = mempty
  115             , _ninjaRules     = mempty
  116             , _ninjaSingles   = mempty
  117             , _ninjaMultiples = mempty
  118             , _ninjaPhonys    = mempty
  119             , _ninjaDefaults  = mempty
  120             , _ninjaPools     = mempty
  121             , _ninjaSpecials  = mempty
  122             }
  123 
  124 -- | The rules defined in a parsed Ninja file.
  125 --
  126 --   @since 0.1.0
  127 {-# INLINE ninjaRules #-}
  128 ninjaRules :: Lens.Lens' (Ninja ann) (HashMap Text (AST.Rule ann))
  129 ninjaRules = Lens.lens _ninjaRules
  130              $ \(MkNinja {..}) x -> MkNinja { _ninjaRules = x, .. }
  131

  180 -- | The usual definition for 'Misc.Annotated'.
  181 --
  182 --   @since 0.1.0
  183 instance Misc.Annotated Ninja where
  184   annotation' f = Lens.lens _ninjaAnn
  185                   $ \(MkNinja {..}) x ->
  186                       MkNinja
  187                       { _ninjaAnn       = x
  188                       , _ninjaRules     = HM.map (fmap f) _ninjaRules
  189                       , _ninjaSingles   = HM.map (fmap f) _ninjaSingles
  190                       , _ninjaMultiples = HM.map (fmap f) _ninjaMultiples
  191                       , .. }
  192 
  193 -- | Converts to
  194 --   @{ann: …, rules: …, singles: …, multiples: …, phonys: …, defaults: …,
  195 --     pools: …, specials: …}@.
  196 --
  197 --   @since 0.1.0
  198 instance (Aeson.ToJSON ann) => Aeson.ToJSON (Ninja ann) where
  199   toJSON (MkNinja {..})
  200     = [ "ann"       .= _ninjaAnn
  201       , "rules"     .= _ninjaRules
  202       , "singles"   .= _ninjaSingles
  203       , "multiples" .= fixMultiples _ninjaMultiples
  204       , "phonys"    .= _ninjaPhonys
  205       , "defaults"  .= _ninjaDefaults
  206       , "pools"     .= _ninjaPools
  207       , "specials"  .= _ninjaSpecials
  208       ] |> Aeson.object
  209     where
  210       fixMultiples :: (Aeson.ToJSON ann)
  211                    => HashMap Outputs (AST.Build ann) -> Aeson.Value
  212       fixMultiples = HM.toList .> map (uncurry printPair) .> Aeson.toJSON
  213 
  214       printPair :: (Aeson.ToJSON ann) => Outputs -> AST.Build ann -> Aeson.Value
  215       printPair outputs build
  216         = Aeson.object ["outputs" .= outputs, "build" .= build]
  217 
  218 -- | Inverse of the 'Aeson.ToJSON' instance.
  219 --
  220 --   @since 0.1.0
  221 instance (Aeson.FromJSON ann) => Aeson.FromJSON (Ninja ann) where
  222   parseJSON = (Aeson.withObject "Ninja" $ \o -> do
  223                   _ninjaAnn       <- (o .: "ann")       >>= pure
  224                   _ninjaRules     <- (o .: "rules")     >>= pure
  225                   _ninjaSingles   <- (o .: "singles")   >>= pure
  226                   _ninjaMultiples <- (o .: "multiples") >>= fixMultiples
  227                   _ninjaPhonys    <- (o .: "phonys")    >>= pure
  228                   _ninjaDefaults  <- (o .: "defaults")  >>= pure
  229                   _ninjaPools     <- (o .: "pools")     >>= pure
  230                   _ninjaSpecials  <- (o .: "specials")  >>= pure
  231                   pure (MkNinja {..}))
  232     where
  233       fixMultiples :: (Aeson.FromJSON ann)
  234                    => Aeson.Value
  235                    -> Aeson.Parser (HashMap Outputs (AST.Build ann))
  236       fixMultiples = Aeson.parseJSON
  237                      >=> mapM parsePair
  238                      >=> (HM.fromList .> pure)
  239 
  240       parsePair :: (Aeson.FromJSON ann)
  241                 => Aeson.Value
  242                 -> Aeson.Parser (Outputs, AST.Build ann)
  243       parsePair = (Aeson.withObject "Ninja.multiples" $ \o -> do
  244                       outputs <- (o .: "outputs") >>= pure
  245                       build   <- (o .: "build")   >>= pure
  246                       pure (outputs, build))

  284 -- | The set of constraints required for a given constraint to be automatically
  285 --   computed for a 'Ninja'.
  286 --
  287 --   @since 0.1.0
  288 type NinjaConstraint (c :: * -> Constraint) (ann :: *)
  289   = ( AST.BuildConstraint c ann
  290     , c (HashMap (HashSet Text) (AST.Build ann))
  291     , c (HashMap Text (HashSet Text))
  292     , c (HashMap Text (AST.Rule ann))
  293     , c (HashMap Text (AST.Build ann))
  294     , c (HashMap Text Int)
  295     , c ann
  296     )
