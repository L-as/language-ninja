   57 
   58 -- | This type represents a POSIX @sh@ command line.
   59 --
   60 --   @since 0.1.0
   61 newtype Command
   62   = MkCommand
   63     { _commandText :: Text
   64     }
   65   deriving ( Eq, Ord, Show, Read, Generic, Hashable, NFData
   66            , Aeson.ToJSON, Aeson.FromJSON )
   67 
   68 -- | Constructor for a 'Command'.
   69 --
   70 --   @since 0.1.0
   71 {-# INLINE makeCommand #-}
   72 makeCommand :: Text -> Command
   73 makeCommand = MkCommand
   74 
   75 -- | An isomorphism between a 'Command' and its underlying 'Text'.
   76 --
   77 --   @since 0.1.0
   78 {-# INLINE commandText #-}
   79 commandText :: Lens.Iso' Command Text
   80 commandText = Lens.iso _commandText MkCommand
   81 
   82 -- | Uses the underlying 'Text' instance.
   83 --
   84 --   @since 0.1.0
   85 instance ( Monad m
   86          , SC.Serial m Text
   87          ) => SC.Serial m Command where
   88   series = SC.newtypeCons MkCommand
   89 
   90 -- | Uses the underlying 'Text' instance.
   91 --
   92 --   @since 0.1.0
   93 instance ( Monad m
   94          , SC.CoSerial m Text
   95          ) => SC.CoSerial m Command where
   96   coseries rs = SC.newtypeAlts rs
   97                 >>- \f -> pure (_commandText .> f)
