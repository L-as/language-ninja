   82 -- | A Ninja @pool@ declaration, as documented
   83 --   <https://ninja-build.org/manual.html#ref_pool here>.
   84 --
   85 --   @since 0.1.0
   86 data Pool
   87   = MkPool
   88     { _poolName  :: !PoolName
   89     , _poolDepth :: !PoolDepth
   90     }
   91   deriving (Eq, Ord, Show, Read, Generic)
   92 
   93 -- | Construct a 'Pool', given its name and depth.
   94 --
   95 --   @since 0.1.0
   96 {-# INLINEABLE makePool #-}
   97 makePool :: PoolName -> PoolDepth -> Maybe Pool
   98 makePool PoolNameDefault    PoolInfinite  = Just makePoolDefault
   99 makePool PoolNameConsole    (PoolDepth 1) = Just makePoolConsole
  100 makePool (PoolNameCustom t) (PoolDepth d) = if d >= 1
  101                                             then Just (makePoolCustom t d)
  102                                             else Nothing
  103 makePool _                  _             = Nothing
  104 
  105 -- | The default pool, i.e.: the one whose name is the empty string.
  106 --
  107 --   @since 0.1.0
  108 {-# INLINE makePoolDefault #-}
  109 makePoolDefault :: Pool
  110 makePoolDefault = MkPool makePoolNameDefault PoolInfinite
  111 
  112 -- | The @console@ pool.
  113 --
  114 --   @since 0.1.0
  115 {-# INLINE makePoolConsole #-}
  116 makePoolConsole :: Pool
  117 makePoolConsole = MkPool makePoolNameConsole (PoolDepth 1)
  118 
  119 -- | Create a pool with the given name and depth.
  120 --
  121 --   @since 0.1.0
  122 {-# INLINE makePoolCustom #-}
  123 makePoolCustom :: Text          -- ^ The pool name.
  124                -> Misc.Positive -- ^ The pool depth.
  125                -> Pool
  126 makePoolCustom name depth = MkPool (makePoolNameCustom name) (PoolDepth depth)
  127 
  128 -- | A 'Getter' that gives the name of a pool.
  129 --
  130 --   @since 0.1.0
  131 {-# INLINE poolName #-}
  132 poolName :: Lens.Getter Pool PoolName
  133 poolName = Lens.to _poolName
  134 
  135 -- | A 'Getter' that gives the depth of a pool.
  136 --
  137 --   @since 0.1.0
  138 {-# INLINE poolDepth #-}
  139 poolDepth :: Lens.Getter Pool PoolDepth
  140 poolDepth = Lens.to _poolDepth
  141 
  142 -- | Converts to @{name: …, depth: …}@.
  143 --
  144 --   @since 0.1.0
  145 instance Aeson.ToJSON Pool where
  146   toJSON (MkPool {..})
  147     = [ "name"  .= _poolName
  148       , "depth" .= _poolDepth
  149       ] |> Aeson.object
  150 
  151 -- | Inverse of the 'Aeson.ToJSON' instance.
  152 --
  153 --   @since 0.1.0
  154 instance Aeson.FromJSON Pool where
  155   parseJSON = (Aeson.withObject "Pool" $ \o -> do
  156                   _poolName  <- (o .: "name")  >>= pure
  157                   _poolDepth <- (o .: "depth") >>= pure
  158                   pure (MkPool {..}))
  159 
  160 -- | Uses the underlying instances.
  161 --
  162 --   @since 0.1.0
  163 instance forall m. (Monad m, SC.Serial m Text) => SC.Serial m Pool where
  164   series = pure makePoolDefault
  165            \/ pure makePoolConsole
  166            \/ (let nameSeries :: SC.Series m Text
  167                    nameSeries = SC.series >>= (\case ""        -> empty
  168                                                      "console" -> empty
  169                                                      x         -> pure x)
  170                in makePoolCustom <$> nameSeries <~> SC.series)
  171 
  172 -- | Uses the underlying instances.
  173 --
  174 --   @since 0.1.0
  175 instance (Monad m, SC.CoSerial m Text) => SC.CoSerial m Pool where
  176   coseries = SC.coseries .> fmap (\f -> convert .> f)
  177     where
  178       convert :: Pool -> (PoolName, PoolDepth)
  179       convert pool = (Lens.view poolName pool, Lens.view poolDepth pool)

  184 instance Hashable Pool

  193 -- | The name of a Ninja pool.
  194 --
  195 --   More information is available
  196 --   <https://ninja-build.org/manual.html#ref_pool here>.
  197 --
  198 --   @since 0.1.0
  199 data PoolName
  200   = PoolNameDefault
  201   | PoolNameConsole
  202   | PoolNameCustom !Text
  203   deriving (Eq, Ord, Show, Read, Generic)
  204 
  205 -- | Create a 'PoolName' corresponding to the built-in default pool, i.e.: the
  206 --   pool that is selected if the @pool@ attribute is set to the empty string.
  207 --
  208 --   @since 0.1.0
  209 {-# INLINE makePoolNameDefault #-}
  210 makePoolNameDefault :: PoolName
  211 makePoolNameDefault = PoolNameDefault
  212 
  213 -- | Create a 'PoolName' corresponding to the built-in @console@ pool.
  214 --
  215 --   @since 0.1.0
  216 {-# INLINE makePoolNameConsole #-}
  217 makePoolNameConsole :: PoolName
  218 makePoolNameConsole = PoolNameConsole
  219 
  220 -- | Create a 'PoolName' corresponding to a custom pool.
  221 --   Note: this can fail at runtime if given the empty string or @"console"@,
  222 --   so you should consider 'parsePoolName' as a safer alternative.
  223 --
  224 --   @since 0.1.0
  225 {-# INLINEABLE makePoolNameCustom #-}
  226 makePoolNameCustom :: Text -> PoolName
  227 makePoolNameCustom ""        = error "Invalid pool name: \"\""
  228 makePoolNameCustom "console" = error "Invalid pool name: \"console\""
  229 makePoolNameCustom text      = PoolNameCustom text
  230 
  231 -- | A one-way prism corresponding to the 'poolNameDefault' constructor.
  232 --
  233 --   @since 0.1.0
  234 {-# INLINE _PoolNameDefault #-}
  235 _PoolNameDefault :: Lens.Getter PoolName (Maybe ())
  236 _PoolNameDefault = Lens.to (\case PoolNameDefault -> Just ()
  237                                   _               -> Nothing)
  238 
  239 -- | A one-way prism corresponding to the 'poolNameConsole' constructor.
  240 --
  241 --   @since 0.1.0
  242 {-# INLINE _PoolNameConsole #-}
  243 _PoolNameConsole :: Lens.Getter PoolName (Maybe ())
  244 _PoolNameConsole = Lens.to (\case PoolNameConsole -> Just ()
  245                                   _               -> Nothing)
  246 
  247 -- | A one-way prism corresponding to the 'poolNameConsole' constructor.
  248 --
  249 --   @since 0.1.0
  250 {-# INLINE _PoolNameCustom #-}
  251 _PoolNameCustom :: Lens.Getter PoolName (Maybe Text)
  252 _PoolNameCustom = Lens.to (\case (PoolNameCustom t) -> Just t
  253                                  _                  -> Nothing)
  254 
  255 -- | An isomorphism between a 'PoolName' and the corresponding 'Text'.
  256 --   Equivalent to @'Lens.iso' 'printPoolName' 'parsePoolName'@.
  257 --
  258 --   @since 0.1.0
  259 {-# INLINE poolNameText #-}
  260 poolNameText :: Lens.Iso' PoolName Text
  261 poolNameText = Lens.iso printPoolName parsePoolName
  262 
  263 -- | Convert a 'PoolName' to the string that, if the @pool@ attribute is set to
  264 --   it, will cause the given 'PoolName' to be parsed.
  265 --
  266 --   >>> printPoolName poolNameDefault
  267 --   ""
  268 --
  269 --   >>> printPoolName poolNameConsole
  270 --   "console"
  271 --
  272 --   >>> printPoolName (poolNameCustom "foobar")
  273 --   "foobar"
  274 --
  275 --   @since 0.1.0
  276 {-# INLINEABLE printPoolName #-}
  277 printPoolName :: PoolName -> Text
  278 printPoolName PoolNameDefault    = ""
  279 printPoolName PoolNameConsole    = "console"
  280 printPoolName (PoolNameCustom t) = t
  281 
  282 -- | Inverse of 'printPoolName'.
  283 --
  284 --   >>> parsePoolName ""
  285 --   PoolNameDefault
  286 --
  287 --   >>> parsePoolName "console"
  288 --   PoolNameConsole
  289 --
  290 --   >>> parsePoolName "foobar"
  291 --   PoolNameCustom "foobar"
  292 --
  293 --   @since 0.1.0
  294 {-# INLINEABLE parsePoolName #-}
  295 parsePoolName :: Text -> PoolName
  296 parsePoolName ""        = makePoolNameDefault
  297 parsePoolName "console" = makePoolNameConsole
  298 parsePoolName t         = makePoolNameCustom t
  299 
  300 -- | Converts from string via 'parsePoolName'.
  301 --
  302 --   @since 0.1.0
  303 instance IsString PoolName where
  304   fromString = Text.pack .> parsePoolName
  305 
  306 -- | Converts to JSON string via 'printPoolName'.
  307 --
  308 --   @since 0.1.0
  309 instance Aeson.ToJSON PoolName where
  310   toJSON = printPoolName .> Aeson.String
  311 
  312 -- | Inverse of the 'Aeson.ToJSON' instance.
  313 --
  314 --   @since 0.1.0
  315 instance Aeson.FromJSON PoolName where
  316   parseJSON = Aeson.withText "PoolName" (parsePoolName .> pure)
  317 
  318 -- | Converts to JSON string via 'printPoolName'.
  319 --
  320 --   @since 0.1.0
  321 instance Aeson.ToJSONKey PoolName where
  322   toJSONKey = Aeson.toJSONKeyText printPoolName
  323 
  324 -- | Inverse of the 'Aeson.ToJSONKey' instance.
  325 --
  326 --   @since 0.1.0
  327 instance Aeson.FromJSONKey PoolName where
  328   fromJSONKey = Aeson.mapFromJSONKeyFunction parsePoolName Aeson.fromJSONKey

  330 -- | Uses the underlying 'Text' instance.
  331 --
  332 --   @since 0.1.0
  333 instance (Monad m, SC.Serial m Text) => SC.Serial m PoolName where

  355 -- | The depth of a Ninja pool.
  356 --
  357 --   More information is available
  358 --   <https://ninja-build.org/manual.html#ref_pool here>.
  359 --
  360 --   @since 0.1.0
  361 data PoolDepth
  362   = PoolDepth !Misc.Positive
  363   | PoolInfinite
  364   deriving (Eq, Ord, Show, Read, Generic)
  365 
  366 -- | Construct a finite 'PoolDepth' from an integer, which should be a number
  367 --   greater than or equal to 1.
  368 --
  369 --   @since 0.1.0
  370 {-# INLINE makePoolDepth #-}
  371 makePoolDepth :: Misc.Positive -> PoolDepth
  372 makePoolDepth = PoolDepth
  373 
  374 -- | Construct an infinite 'PoolDepth'. This constructor is needed for the
  375 --   default pool (@pool = ""@), which has an infinite depth.
  376 --
  377 --   @since 0.1.0
  378 {-# INLINE makePoolInfinite #-}
  379 makePoolInfinite :: PoolDepth
  380 makePoolInfinite = PoolInfinite
  381 
  382 -- | An isomorphism between a 'PoolDepth' and a @'Maybe' 'Misc.Positive'@;
  383 --   the 'Nothing' case maps to 'makePoolInfinite' and the 'Just' case
  384 --   maps to 'makePoolDepth'.
  385 --
  386 --   @since 0.1.0
  387 {-# INLINE poolDepthPositive #-}
  388 poolDepthPositive :: Lens.Iso' PoolDepth (Maybe Misc.Positive)
  389 poolDepthPositive = Lens.iso fromPD toPD
  390   where
  391     {-# INLINE fromPD #-}
  392     fromPD :: PoolDepth -> Maybe Misc.Positive
  393     fromPD (PoolDepth p) = Just p
  394     fromPD PoolInfinite  = Nothing
  395 
  396     {-# INLINE toPD #-}
  397     toPD :: Maybe Misc.Positive -> PoolDepth
  398     toPD (Just p) = PoolDepth p
  399     toPD Nothing  = PoolInfinite
  400 
  401 -- | Converts 'makePoolInfinite' to @"infinite"@ and 'makePoolDepth' to the
  402 --   corresponding JSON number.
  403 --
  404 --   @since 0.1.0
  405 instance Aeson.ToJSON PoolDepth where
  406   toJSON (PoolDepth i) = Aeson.toJSON i
  407   toJSON PoolInfinite  = "infinite"
  408 
  409 -- | Inverse of the 'Aeson.ToJSON' instance.
  410 --
  411 --   @since 0.1.0
  412 instance Aeson.FromJSON PoolDepth where
  413   parseJSON (v@(Aeson.Number _))      = PoolDepth <$> Aeson.parseJSON v
  414   parseJSON (Aeson.String "infinite") = pure PoolInfinite
  415   parseJSON owise                     = Aeson.typeMismatch "PoolDepth" owise
  416 
  417 -- | Default 'SC.Serial' instance via 'Generic'.
  418 --
  419 --   @since 0.1.0
  420 instance (Monad m) => SC.Serial m PoolDepth where
  421   series = pure PoolInfinite
  422            \/ (SC.series |> fmap PoolDepth)
  423 
  424 -- | Default 'SC.CoSerial' instance via 'Generic'.
  425 --
  426 --   @since 0.1.0
  427 instance (Monad m) => SC.CoSerial m PoolDepth where
  428   coseries = SC.coseries
  429              .> fmap (\f -> \case (PoolDepth i) -> f (Just i)
  430                                   PoolInfinite  -> f Nothing)
  431 
  432 -- | Default 'Hashable' instance via 'Generic'.
  433 --
  434 --   @since 0.1.0
  435 instance Hashable PoolDepth
  436 
  437 -- | Default 'NFData' instance via 'Generic'.
  438 --
  439 --   @since 0.1.0
  440 instance NFData PoolDepth
