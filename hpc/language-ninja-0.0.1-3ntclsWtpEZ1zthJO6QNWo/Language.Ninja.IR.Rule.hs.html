   74 -- | A Ninja @rule@ declaration, as documented
   75 --   <https://ninja-build.org/manual.html#_rules here>.
   76 --
   77 --   @since 0.1.0
   78 data Rule
   79   = MkRule
   80     { _ruleName         :: !Text
   81     , _ruleCommand      :: !Command
   82     , _ruleDescription  :: !(Maybe Text)
   83     , _rulePool         :: !PoolName
   84     , _ruleDepfile      :: !(Maybe Path)
   85     , _ruleSpecialDeps  :: !(Maybe SpecialDeps)
   86     , _ruleGenerator    :: !Bool
   87     , _ruleRestat       :: !Bool
   88     , _ruleResponseFile :: !(Maybe ResponseFile)
   89     }
   90   deriving (Eq, Ord, Show, Generic)
   91 
   92 -- | Construct an 'Rule' with the given name and command, with default values
   93 --   for all other attributes (e.g.: 'False', 'Nothing', 'poolDefault').
   94 --
   95 --   @since 0.1.0
   96 {-# INLINE makeRule #-}
   97 makeRule :: Text
   98          -- ^ The rule name.
   99          -> Command
  100          -- ^ The command to run.
  101          -> Rule
  102          -- ^ A rule that runs this command.
  103 makeRule name cmd
  104   = MkRule
  105     { _ruleName         = name
  106     , _ruleCommand      = cmd
  107     , _ruleDescription  = Nothing
  108     , _rulePool         = makePoolNameDefault
  109     , _ruleDepfile      = Nothing
  110     , _ruleSpecialDeps  = Nothing
  111     , _ruleGenerator    = False
  112     , _ruleRestat       = False
  113     , _ruleResponseFile = Nothing
  114     }
  115 
  116 -- | The name of the rule.
  117 --
  118 --   @since 0.1.0
  119 {-# INLINE ruleName #-}
  120 ruleName :: Lens.Lens' Rule Text
  121 ruleName = Lens.lens _ruleName
  122            $ \(MkRule {..}) x -> MkRule { _ruleName = x, .. }
  123 
  124 -- | The command that this rule will run.
  125 --
  126 --   @since 0.1.0
  127 {-# INLINE ruleCommand #-}
  128 ruleCommand :: Lens.Lens' Rule Command
  129 ruleCommand = Lens.lens _ruleCommand
  130               $ \(MkRule {..}) x -> MkRule { _ruleCommand = x, .. }
  131 
  132 -- | A short description of the command, used to pretty-print the command
  133 --   as it's running. The @ninja -v@ flag controls whether to print the
  134 --   full command or its description; if a command fails, the full command
  135 --   line will always be printed before the command's output.
  136 --
  137 --   @since 0.1.0
  138 {-# INLINE ruleDescription #-}
  139 ruleDescription :: Lens.Lens' Rule (Maybe Text)
  140 ruleDescription = Lens.lens _ruleDescription
  141                   $ \(MkRule {..}) x -> MkRule { _ruleDescription = x, .. }
  142 
  143 -- | The process pool in which this rule will be executed.
  144 --
  145 --   @since 0.1.0
  146 {-# INLINE rulePool #-}
  147 rulePool :: Lens.Lens' Rule PoolName
  148 rulePool = Lens.lens _rulePool
  149            $ \(MkRule {..}) x -> MkRule { _rulePool = x, .. }
  150 
  151 -- | If set, this should be a path to an optional Makefile that contains
  152 --   extra implicit dependencies. This is used to support C/C++ header
  153 --   dependencies. For more information, read the Ninja documentation
  154 --   <https://ninja-build.org/manual.html#_depfile here>.
  155 --
  156 --   @since 0.1.0
  157 {-# INLINE ruleDepfile #-}
  158 ruleDepfile :: Lens.Lens' Rule (Maybe Path)
  159 ruleDepfile = Lens.lens _ruleDepfile
  160               $ \(MkRule {..}) x -> MkRule { _ruleDepfile = x, .. }
  161 
  162 -- | If set, enables special dependency processing used in C/C++ header
  163 --   dependencies. For more information, read the Ninja documentation
  164 --   <https://ninja-build.org/manual.html#_deps here>.
  165 --
  166 --   @since 0.1.0
  167 {-# INLINE ruleSpecialDeps #-}
  168 ruleSpecialDeps :: Lens.Lens' Rule (Maybe SpecialDeps)
  169 ruleSpecialDeps = Lens.lens _ruleSpecialDeps
  170                   $ \(MkRule {..}) x -> MkRule { _ruleSpecialDeps = x, .. }
  171 
  172 -- | If this is true, specifies that this rule is used to re-invoke the
  173 --   generator program. Files built using generator rules are treated
  174 --   specially in two ways: firstly, they will not be rebuilt if the
  175 --   command line changes; and secondly, they are not cleaned by default.
  176 --
  177 --   @since 0.1.0
  178 {-# INLINE ruleGenerator #-}
  179 ruleGenerator :: Lens.Lens' Rule Bool
  180 ruleGenerator = Lens.lens _ruleGenerator
  181                 $ \(MkRule {..}) x -> MkRule { _ruleGenerator = x, .. }
  182 
  183 -- | If true, causes Ninja to re-stat the command's outputs after
  184 --   execution of the command. Each output whose modification time the
  185 --   command did not change will be treated as though it had never needed
  186 --   to be built. This may cause the output's reverse dependencies to be
  187 --   removed from the list of pending build actions.
  188 --
  189 --   @since 0.1.0
  190 {-# INLINE ruleRestat #-}
  191 ruleRestat :: Lens.Lens' Rule Bool
  192 ruleRestat = Lens.lens _ruleRestat
  193              $ \(MkRule {..}) x -> MkRule { _ruleRestat = x, .. }
  194 
  195 -- | If present, Ninja will use a response file for the given command,
  196 --   i.e. write the selected string to the given file before calling the
  197 --   command and delete the file after the command is done.
  198 --
  199 --   This is particularly useful on Windows OS, where the maximal length
  200 --   of a command line is limited and response files must be used instead.
  201 --
  202 --   @since 0.1.0
  203 {-# INLINE ruleResponseFile #-}
  204 ruleResponseFile :: Lens.Lens' Rule (Maybe ResponseFile)
  205 ruleResponseFile = Lens.lens _ruleResponseFile
  206                    $ \(MkRule {..}) x -> MkRule { _ruleResponseFile = x, .. }
  207 
  208 -- | Converts to
  209 --   @{name: …, command: …, desc: …, pool: …, depfile: …,
  210 --     deps: …, generator: …, restat: …, rsp: …}@.
  211 --
  212 --   @since 0.1.0
  213 instance Aeson.ToJSON Rule where
  214   toJSON (MkRule {..})
  215     = [ "name"      .= _ruleName
  216       , "command"   .= _ruleCommand
  217       , "desc"      .= _ruleDescription
  218       , "pool"      .= _rulePool
  219       , "depfile"   .= _ruleDepfile
  220       , "deps"      .= _ruleSpecialDeps
  221       , "generator" .= _ruleGenerator
  222       , "restat"    .= _ruleRestat
  223       , "rsp"       .= _ruleResponseFile
  224       ] |> Aeson.object
  225 
  226 -- | Inverse of the 'Aeson.ToJSON' instance.
  227 --
  228 --   @since 0.1.0
  229 instance Aeson.FromJSON Rule where
  230   parseJSON = (Aeson.withObject "Rule" $ \o -> do
  231                   _ruleName         <- (o .: "name")      >>= pure
  232                   _ruleCommand      <- (o .: "command")   >>= pure
  233                   _ruleDescription  <- (o .: "desc")      >>= pure
  234                   _rulePool         <- (o .: "pool")      >>= pure
  235                   _ruleDepfile      <- (o .: "depfile")   >>= pure
  236                   _ruleSpecialDeps  <- (o .: "deps")      >>= pure
  237                   _ruleGenerator    <- (o .: "generator") >>= pure
  238                   _ruleRestat       <- (o .: "restat")    >>= pure
  239                   _ruleResponseFile <- (o .: "rsp")       >>= pure
  240                   pure (MkRule {..}))
  241

  268 -- | Special dependency information, as described
  269 --   <https://ninja-build.org/manual.html#ref_headers here>.
  270 --
  271 --   @since 0.1.0
  272 data SpecialDeps
  273   = SpecialDepsGCC
  274   | SpecialDepsMSVC !Text
  275   deriving (Eq, Ord, Show, Read, Generic)
  276 
  277 -- | Construct a 'SpecialDeps' corresponding to the case in which @deps = gcc@
  278 --   is set in a Ninja build rule.
  279 --
  280 --   @since 0.1.0
  281 {-# INLINE makeSpecialDepsGCC #-}
  282 makeSpecialDepsGCC :: SpecialDeps
  283 makeSpecialDepsGCC = SpecialDepsGCC
  284 
  285 -- | Construct a 'SpecialDeps' corresponding to the case in which @deps = msvc@
  286 --   is set and @msvc_deps_prefix = …@.
  287 --
  288 --   The @msvc_deps_prefix@ field defines the string which should be stripped
  289 --   from @msvc@'s @/showIncludes@ output. It is only needed if the version of
  290 --   Visual Studio being used is not English. The value of @msvc_deps_prefix@
  291 --   is @"Note: including file: "@ by default.
  292 --
  293 --   @since 0.1.0
  294 {-# INLINE makeSpecialDepsMSVC #-}
  295 makeSpecialDepsMSVC :: Text
  296                     -> SpecialDeps
  297 makeSpecialDepsMSVC = SpecialDepsMSVC
  298 
  299 -- | A prism for the @deps = gcc@ case.
  300 --
  301 --   @since 0.1.0
  302 {-# INLINE _SpecialDepsGCC #-}
  303 _SpecialDepsGCC :: Lens.Prism' SpecialDeps ()
  304 _SpecialDepsGCC = Lens.prism (const makeSpecialDepsGCC)
  305                   $ \case SpecialDepsGCC -> Right ()
  306                           owise          -> Left owise
  307 
  308 -- | A prism for the @deps = msvc@ / @msvc_deps_prefix = …@ case.
  309 --
  310 --   @since 0.1.0
  311 {-# INLINE _SpecialDepsMSVC #-}
  312 _SpecialDepsMSVC :: Lens.Prism' SpecialDeps Text
  313 _SpecialDepsMSVC = Lens.prism makeSpecialDepsMSVC
  314                    $ \case (SpecialDepsMSVC prefix) -> Right prefix
  315                            owise                    -> Left owise
  316 
  317 -- | Converts to @{deps: "gcc"}@ or @{deps: "msvc", prefix: …}@.
  318 --
  319 --   @since 0.1.0
  320 instance Aeson.ToJSON SpecialDeps where
  321   toJSON = go
  322     where
  323       go SpecialDepsGCC      = Aeson.object ["deps" .= gcc]
  324       go (SpecialDepsMSVC p) = Aeson.object ["deps" .= msvc, "prefix" .= p]
  325 
  326       (gcc, msvc) = ("gcc", "msvc") :: (Aeson.Value, Aeson.Value)
  327 
  328 -- | Inverse of the 'Aeson.ToJSON' instance.
  329 --
  330 --   @since 0.1.0
  331 instance Aeson.FromJSON SpecialDeps where
  332   parseJSON = Aeson.withObject "SpecialDeps" $ \o -> do
  333     deps <- o .: "deps"
  334     case Text.pack deps of
  335       "gcc"  -> pure SpecialDepsGCC
  336       "msvc" -> SpecialDepsMSVC <$> (o .: "prefix")
  337       owise  -> [ "Invalid deps type ", "\"", owise, "\"; "
  338                 , "should be one of [\"gcc\", \"msvc\"]."
  339                 ] |> mconcat |> Text.unpack |> fail
  340 
  341 -- | Default 'Hashable' instance via 'Generic'.
  342 --
  343 --   @since 0.1.0
  344 instance Hashable SpecialDeps

  367 -- | A response file to use during rule execution, as documented
  368 --   <https://ninja-build.org/manual.html#ref_rule here>.
  369 --
  370 --   @since 0.1.0
  371 data ResponseFile
  372   = MkResponseFile
  373     { _responseFilePath    :: !Path
  374     , _responseFileContent :: !Text
  375     }
  376   deriving (Eq, Ord, Show, Generic)
  377 
  378 -- | Construct a 'ResponseFile' with the given 'Path' and content 'Text'.
  379 --
  380 --   @since 0.1.0
  381 {-# INLINE makeResponseFile #-}
  382 makeResponseFile :: Path
  383                  -- ^ Corresponds to @rspfile@.
  384                  -> Text
  385                  -- ^ Corresponds to @rspfile_content@.
  386                  -> ResponseFile
  387 makeResponseFile = MkResponseFile
  388 
  389 -- | A lens for the @rspfile@ field.
  390 --
  391 --   @since 0.1.0
  392 {-# INLINE responseFilePath #-}
  393 responseFilePath :: Lens.Lens' ResponseFile Path
  394 responseFilePath = Lens.lens _responseFilePath
  395                    $ \(MkResponseFile {..}) x ->
  396                        MkResponseFile { _responseFilePath = x, .. }
  397 
  398 -- | A lens for the @rspfile_content@ field.
  399 --
  400 --   @since 0.1.0
  401 {-# INLINE responseFileContent #-}
  402 responseFileContent :: Lens.Lens' ResponseFile Text
  403 responseFileContent = Lens.lens _responseFileContent
  404                       $ \(MkResponseFile {..}) x ->
  405                           MkResponseFile { _responseFileContent = x, .. }
  406 
  407 -- | Converts to @{path: …, content: …}@.
  408 --
  409 --   @since 0.1.0
  410 instance Aeson.ToJSON ResponseFile where
  411   toJSON (MkResponseFile {..})
  412     = [ "path"    .= _responseFilePath
  413       , "content" .= _responseFileContent
  414       ] |> Aeson.object
  415 
  416 -- | Inverse of the 'Aeson.ToJSON' instance.
  417 --
  418 --   @since 0.1.0
  419 instance Aeson.FromJSON ResponseFile where
  420   parseJSON = Aeson.withObject "ResponseFile"
  421               $ \o -> MkResponseFile <$> (o .: "path") <*> (o .: "content")
  422
