   57 -- | This type represents a positive number; i.e.: an integer greater than zero.
   58 --
   59 --   @since 0.1.0
   60 newtype Positive
   61   = MkPositive
   62     { _fromPositive :: Int
   63     }
   64   deriving ( Eq, Ord, Real, Integral, Enum, Show, Read, Generic
   65            , Hashable, NFData
   66            , Aeson.ToJSON, Aeson.FromJSON )
   67 
   68 -- | This instance uses 'error' to preserve the 'Positive' invariant.
   69 --
   70 --   @since 0.1.0
   71 instance Num Positive where
   72   (MkPositive a) + (MkPositive b) = MkPositive (a + b)
   73   (MkPositive a) * (MkPositive b) = MkPositive (a * b)
   74   abs = id
   75   signum = const 1
   76   negate = ["Prelude.negate: Positive cannot be negated"]
   77            |> mconcat |> error
   78   fromInteger i | i > 0     = MkPositive (fromIntegral i)
   79                 | otherwise = [ "Prelude.fromInteger: "
   80                               , "Positive invariant violated"
   81                               ] |> mconcat |> error
   82 
   83 -- | Constructor for a 'Positive'.
   84 --
   85 --   @since 0.1.0
   86 {-# INLINEABLE makePositive #-}
   87 makePositive :: Int -> Maybe Positive
   88 makePositive i | i > 0     = Just (MkPositive i)
   89                | otherwise = Nothing
   90 
   91 -- | A 'Lens.Getter' for the 'Int' underlying a 'Positive'.
   92 --
   93 --   @since 0.1.0
   94 {-# INLINE fromPositive #-}
   95 fromPositive :: Lens.Getter Positive Int
   96 fromPositive = Lens.to _fromPositive
   97 
   98 -- | Uses the underlying 'Int' instance.
   99 --
  100 --   @since 0.1.0
  101 instance (Monad m) => SC.Serial m Positive where
  102   series = MkPositive <$> (SC.series `suchThat` (> 0))
  103     where
  104       suchThat :: SC.Series m a -> (a -> Bool) -> SC.Series m a
  105       suchThat s p = s >>= \x -> if p x then pure x else empty
  106 
  107 -- | Uses the underlying 'Int' instance.
  108 --
  109 --   @since 0.1.0
  110 instance (Monad m) => SC.CoSerial m Positive where
  111   coseries = SC.coseries .> fmap (\f -> _fromPositive .> f)
