   66 -- | Pretty-print a 'AST.Ninja'.
   67 --
   68 --   @since 0.1.0
   69 prettyNinja :: AST.Ninja () -> Text
   70 prettyNinja ninja
   71   = [ map prettyRule     (HM.toList (Lens.view AST.ninjaRules     ninja))
   72     , map prettySingle   (HM.toList (Lens.view AST.ninjaSingles   ninja))
   73     , map prettyMultiple (HM.toList (Lens.view AST.ninjaMultiples ninja))
   74     , map prettyPhony    (HM.toList (Lens.view AST.ninjaPhonys    ninja))
   75     , map prettyDefault  (HS.toList (Lens.view AST.ninjaDefaults  ninja))
   76     , map prettyPool     (HM.toList (Lens.view AST.ninjaPools     ninja))
   77     ] |> mconcat |> mconcat
   78 
   79 -- | Pretty-print an 'AST.Expr'
   80 --
   81 --   @since 0.1.0
   82 prettyExpr :: AST.Expr () -> Text
   83 prettyExpr = go .> mconcat
   84   where
   85     go (AST.Exprs _   es) = map prettyExpr es
   86     go (AST.Lit   _ text) = [text]
   87     go (AST.Var   _ name) = ["${", name, "}"]
   88 
   89 -- | Pretty-print a Ninja @rule@ declaration.
   90 --
   91 --   @since 0.1.0
   92 prettyRule :: (Text, AST.Rule ()) -> Text
   93 prettyRule (name, rule) = do
   94   let binds = Lens.view AST.ruleBind rule
   95               |> HM.toList
   96               |> map (second prettyExpr .> prettyBind)
   97               |> mconcat
   98   mconcat ["rule ", name, "\n", binds]
   99 
  100 -- | Pretty-print a Ninja @build@ declaration with one output.
  101 --
  102 --   @since 0.1.0
  103 prettySingle :: (Text, AST.Build ()) -> Text
  104 prettySingle (output, build) = prettyMultiple (HS.singleton output, build)
  105 
  106 -- | Pretty-print a Ninja @build@ declaration with multiple outputs.
  107 --
  108 --   @since 0.1.0
  109 prettyMultiple :: (HashSet Text, AST.Build ()) -> Text
  110 prettyMultiple (outputs, build) = do
  111   let prefixIfThere :: Text -> Text -> Text
  112       prefixIfThere pfx rest = if Text.all isSpace rest then "" else pfx <> rest
  113 
  114   let unwordsSet :: HashSet Text -> Text
  115       unwordsSet = HS.toList .> Text.unwords
  116 
  117   let ruleName  = Lens.view AST.buildRule build
  118   let deps      = Lens.view AST.buildDeps build
  119   let binds     = Lens.view AST.buildBind build
  120   let normal    = Lens.view AST.depsNormal    deps
  121   let implicit  = Lens.view AST.depsImplicit  deps
  122   let orderOnly = Lens.view AST.depsOrderOnly deps
  123 
  124   mconcat
  125     [ "build ", Text.unwords (HS.toList outputs), ": "
  126     , ruleName, " ", unwordsSet normal
  127     , prefixIfThere " | "  (unwordsSet implicit)
  128     , prefixIfThere " || " (unwordsSet orderOnly), "\n"
  129     , HM.toList binds |> map prettyBind |> mconcat
  130     ]
  131 
  132 -- | Pretty-print a Ninja phony @build@ declaration.
  133 --
  134 --   @since 0.1.0
  135 prettyPhony :: (Text, HashSet Text) -> Text
  136 prettyPhony (name, inputs)
  137   = [ ["build ", name, ": phony ", Text.unwords (HS.toList inputs)]
  138     ] |> map mconcat |> Text.unlines
  139 
  140 -- | Pretty-print a Ninja @default@ declaration.
  141 --
  142 --   @since 0.1.0
  143 prettyDefault :: Text -> Text
  144 prettyDefault target
  145   = [ ["default ", target]
  146     ] |> map mconcat |> Text.unlines
  147 
  148 -- | Pretty-print a Ninja @pool@ declaration.
  149 --
  150 --   @since 0.1.0
  151 prettyPool :: (Text, Int) -> Text
  152 prettyPool (name, depth)
  153   = [ ["pool ", name]
  154     , ["    depth = ", tshow depth]
  155     ] |> map mconcat |> Text.unlines
  156 
  157 -- | Pretty-print a Ninja indented binding.
  158 --
  159 --   @since 0.1.0
  160 prettyBind :: (Text, Text) -> Text
  161 prettyBind (name, value) = mconcat ["    ", name, " = ", value, "\n"]
  162

  165 tshow :: (Show s) => s -> Text
  166 tshow = show .> Text.pack
