   58 -- | The type of errors encountered during parsing.
   59 --
   60 --   @since 0.1.0
   61 data ParseError
   62   = -- | Generic catch-all error constructor. Avoid using this.
   63     --
   64     --   @since 0.1.0
   65     GenericParseError      !Text
   66   | -- | @Lexer failed at binding: <text>@
   67     --
   68     --   @since 0.1.0
   69     LexBindingFailure      !Text
   70   | -- | @Expected a colon@
   71     --
   72     --   @since 0.1.0
   73     LexExpectedColon
   74   | -- | @Unexpected $ followed by unexpected stuff@
   75     --
   76     --   @since 0.1.0
   77     LexUnexpectedDollar
   78   | -- | Lexer expected a separator character but found something else
   79     --
   80     --   @since 0.1.0
   81     LexUnexpectedSeparator Char
   82   | -- | Any other lexer error.
   83     --
   84     --   @since 0.1.0
   85     LexParsecError         !(M.ParseError Char M.Dec)
   86   | -- | @Could not parse depth field in pool, got: <text>@
   87     --
   88     --   @since 0.1.0
   89     ParseBadDepthField     !Text
   90   | -- | @Unexpected binding defining <text>@
   91     --
   92     --   @since 0.1.0
   93     ParseUnexpectedBinding !Text
   94   deriving (Eq, Show, Generic)
   95 
   96 -- FIXME: remove unused errors here
   97 
   98 -- | Default instance.
   99 --
  100 --   @since 0.1.0
  101 instance Exception ParseError
  102 
  103 -- | Throw a 'ParseError'.
  104 --
  105 --   @since 0.1.0
  106 throwParseError :: (MonadError ParseError m) => ParseError -> m a
  107 throwParseError = throwError
  108 
  109 -- | Throw a generic catch-all 'ParseError'.
  110 --
  111 --   @since 0.1.0
  112 throwGenericParseError :: (MonadError ParseError m) => Text -> m a
  113 throwGenericParseError msg = throwParseError (GenericParseError msg)
  114 
  115 -- | Throw a 'LexBindingFailure' error.
  116 --
  117 --   @since 0.1.0
  118 throwLexBindingFailure :: (MonadError ParseError m) => Text -> m a
  119 throwLexBindingFailure t = throwParseError (LexBindingFailure t)
  120 
  121 -- | Throw a 'LexExpectedColon' error.
  122 --
  123 --   @since 0.1.0
  124 throwLexExpectedColon :: (MonadError ParseError m) => m a
  125 throwLexExpectedColon = throwParseError LexExpectedColon
  126 
  127 -- | Throw a 'LexUnexpectedColon' error.
  128 --
  129 --   @since 0.1.0
  130 throwLexUnexpectedDollar :: (MonadError ParseError m) => m a
  131 throwLexUnexpectedDollar = throwParseError LexUnexpectedDollar
  132 
  133 -- | Throw a 'LexUnexpectedSeparator' error.
  134 --
  135 --   @since 0.1.0
  136 throwLexUnexpectedSeparator :: (MonadError ParseError m) => Char -> m a
  137 throwLexUnexpectedSeparator c = throwParseError (LexUnexpectedSeparator c)
  138 
  139 -- | Throw a 'LexParsecError' error.
  140 --
  141 --   @since 0.1.0
  142 throwLexParsecError :: (MonadError ParseError m)
  143                     => M.ParseError Char M.Dec -> m a
  144 throwLexParsecError pe = throwParseError (LexParsecError pe)
  145 
  146 -- | Throw a 'ParseBadDepthField' error.
  147 --
  148 --   @since 0.1.0
  149 throwParseBadDepthField :: (MonadError ParseError m) => Text -> m a
  150 throwParseBadDepthField t = throwParseError (ParseBadDepthField t)
  151 
  152 -- | Throw a 'ParseUnexpectedBinding' error.
  153 --
  154 --   @since 0.1.0
  155 throwParseUnexpectedBinding :: (MonadError ParseError m) => Text -> m a
  156 throwParseUnexpectedBinding t = throwParseError (ParseUnexpectedBinding t)
