   67 -- | Ninja top-level metadata, as documented
   68 --   <https://ninja-build.org/manual.html#ref_toplevel here>.
   69 --
   70 --   @since 0.1.0
   71 data Meta
   72   = MkMeta
   73     { _metaReqVersion :: !(Maybe Ver.Version)
   74     , _metaBuildDir   :: !(Maybe Path)
   75     }
   76   deriving (Eq, Ord, Show, Generic)
   77 
   78 -- | Construct a default 'Meta' value.
   79 --
   80 --   @since 0.1.0
   81 {-# INLINE makeMeta #-}
   82 makeMeta :: Meta
   83 makeMeta = MkMeta
   84            { _metaReqVersion = Nothing
   85            , _metaBuildDir   = Nothing
   86            }
   87 
   88 -- | Corresponds to the @ninja_required_version@ top-level variable.
   89 --
   90 --   @since 0.1.0
   91 {-# INLINE metaReqVersion #-}
   92 metaReqVersion :: Lens.Lens' Meta (Maybe Ver.Version)
   93 metaReqVersion = Lens.lens _metaReqVersion
   94                  $ \(MkMeta {..}) x -> MkMeta { _metaReqVersion = x, .. }
   95 
   96 -- | Corresponds to the @builddir@ top-level variable.
   97 --
   98 --   @since 0.1.0
   99 {-# INLINE metaBuildDir #-}
  100 metaBuildDir :: Lens.Lens' Meta (Maybe Path)
  101 metaBuildDir = Lens.lens _metaBuildDir
  102                $ \(MkMeta {..}) x -> MkMeta { _metaBuildDir = x, .. }
  103 
  104 -- | Converts to @{req-version: …, build-dir: …}@.
  105 --
  106 --   @since 0.1.0
  107 instance Aeson.ToJSON Meta where
  108   toJSON (MkMeta {..})
  109     = [ "req-version" .= fmap versionJ _metaReqVersion
  110       , "build-dir"   .= _metaBuildDir
  111       ] |> Aeson.object
  112     where
  113       versionJ :: Ver.Version -> Aeson.Value
  114       versionJ = Ver.prettyVer .> Aeson.toJSON
  115 
  116 -- | Inverse of the 'Aeson.ToJSON' instance.
  117 --
  118 --   @since 0.1.0
  119 instance Aeson.FromJSON Meta where
  120   parseJSON = (Aeson.withObject "Meta" $ \o -> do
  121                   _metaReqVersion <- (o .: "req-version") >>= maybeVersionP
  122                   _metaBuildDir   <- (o .: "build-dir")   >>= pure
  123                   pure (MkMeta {..}))
  124     where
  125       maybeVersionP :: Maybe Aeson.Value -> Aeson.Parser (Maybe Ver.Version)
  126       maybeVersionP = fmap versionP .> sequenceA
  127 
  128       versionP :: Aeson.Value -> Aeson.Parser Ver.Version
  129       versionP = Aeson.withText "Version" (megaparsecToAeson Ver.version')
  130

  134 instance Hashable Meta

  159 -- This function converts a @megaparsec@ parser to an @aeson@ parser.
  160 -- Mainly, it handles converting the error output from @megaparsec@ to a
  161 -- string that is appropriate for 'fail'.
  162 megaparsecToAeson :: Mega.Parsec Mega.Dec Text t
  163                   -> (Text -> Aeson.Parser t)
  164 megaparsecToAeson parser text = case Mega.runParser parser "" text of
  165                                   Left  e -> fail (Mega.parseErrorPretty e)
  166                                   Right x -> pure x
