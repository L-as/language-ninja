   95 -- | A Ninja-style environment, basically a nonempty list of hash tables.
   96 --
   97 --   @since 0.1.0
   98 newtype Env k v
   99   = MkEnv
  100     { _fromEnv :: Maps k v
  101     }
  102   deriving (Eq, Show, Generic)
  103 
  104 -- | Construct an empty environment.
  105 --
  106 --   @since 0.1.0
  107 {-# INLINE makeEnv #-}
  108 makeEnv :: Env k v
  109 makeEnv = MkEnv (HM.empty :| [])
  110 
  111 -- | An isomorphism between an 'Env' and a nonempty list of 'HashMap's.
  112 --
  113 --   @since 0.1.0
  114 {-# INLINE fromEnv #-}
  115 fromEnv :: Lens.Iso' (Env k v) (Maps k v)
  116 fromEnv = Lens.iso _fromEnv MkEnv
  117 
  118 -- | Get the first 'HashMap' in the underlying nonempty list.
  119 --
  120 --   @since 0.1.0
  121 {-# INLINEABLE headEnv #-}
  122 headEnv :: Env k v -> HashMap k v
  123 headEnv (MkEnv (m :| _)) = m
  124 
  125 -- | If the remainder of the underlying nonempty list is nonempty, return
  126 --   the remainder after 'Env' wrapping. Otherwise, return 'Nothing'.
  127 --
  128 --   @since 0.1.0
  129 {-# INLINEABLE tailEnv #-}
  130 tailEnv :: Env k v -> Maybe (Env k v)
  131 tailEnv (MkEnv (_ :| e)) = MkEnv <$> NE.nonEmpty e
  132 
  133 -- | Push a new 'Env' onto the stack.
  134 --
  135 --   @since 0.1.0
  136 {-# INLINEABLE scopeEnv #-}
  137 scopeEnv :: Env k v -> Env k v
  138 scopeEnv e = MkEnv (NE.cons HM.empty (_fromEnv e))
  139 
  140 -- | Add the given key and value to the given 'Env'.
  141 --
  142 --   @since 0.1.0
  143 {-# INLINEABLE addEnv #-}
  144 addEnv :: (Key k) => k -> v -> Env k v -> Env k v
  145 addEnv k v (MkEnv (m :| rest)) = MkEnv (HM.insert k v m :| rest)
  146 
  147 -- | Look up the given key in the given 'Env'.
  148 --
  149 --   @since 0.1.0
  150 {-# INLINEABLE askEnv #-}
  151 askEnv :: (Key k) => Env k v -> k -> Maybe v
  152 askEnv env k = HM.lookup k (headEnv env)
  153                <|> (tailEnv env >>= (`askEnv` k))
  154 
  155 -- | Converts to a (nonempty) array of JSON objects.
  156 --
  157 --   @since 0.1.0
  158 instance (Aeson.ToJSONKey k, Aeson.ToJSON v) => Aeson.ToJSON (Env k v) where
  159   toJSON = _fromEnv .> NE.toList .> Aeson.toJSON
  160 
  161 -- | Inverse of the 'Aeson.ToJSON' instance.
  162 --
  163 --   @since 0.1.0
  164 instance ( Key k, Aeson.FromJSONKey k, Aeson.FromJSON v
  165          ) => Aeson.FromJSON (Env k v) where
  166   parseJSON = Aeson.parseJSON
  167               >=> NE.nonEmpty
  168               .>  maybe (fail "Env list was empty!") pure
  169               .>  fmap MkEnv
  170 
  171 -- | Reasonable 'QC.Arbitrary' instance for 'Env'.
  172 --
  173 --   @since 0.1.0
  174 instance ( Key k, QC.Arbitrary k, QC.Arbitrary v
  175          ) => QC.Arbitrary (Env k v) where
  176   arbitrary = QC.arbitrary
  177               |> fmap (map (uncurry addEnv .> Endo)
  178                        .> mconcat
  179                        .> (\e -> appEndo e makeEnv))
  180 
  181 -- | Default 'Hashable' instance via 'Generic'.
  182 --
  183 --   @since 0.1.0
  184 instance (Hashable k, Hashable v) => Hashable (Env k v)
  185 
  186 -- | Default 'NFData' instance via 'Generic'.
  187 --
  188 --   @since 0.1.0
  189 instance (NFData k, NFData v) => NFData (Env k v)
  190 
  191 -- | Uses the underlying 'Maps' instance.
  192 --
  193 --   @since 0.1.0
  194 instance ( Monad m, EnvConstraint (SC.Serial m) k v
  195          ) => SC.Serial m (Env k v) where
  196   series = SC.series |> fmap MkEnv
  197 
  198 -- | Uses the underlying 'Maps' instance.
  199 --
  200 --   @since 0.1.0
  201 instance ( Monad m, EnvConstraint (SC.CoSerial m) k v
  202          ) => SC.CoSerial m (Env k v) where
  203   coseries = SC.coseries .> fmap (\f -> _fromEnv .> f)
  204 
  205 -- | The set of constraints required for a given constraint to be automatically
  206 --   computed for an 'Env'.
  207 --
  208 --   @since 0.1.0
  209 type EnvConstraint (c :: * -> Constraint) (k :: *) (v :: *)
  210   = (Key k, c k, c v, c (Maps k v))
  211

  214 -- | A constraint alias for @('Eq' k, 'Hashable' k)@.
  215 --
  216 --   @since 0.1.0
  217 type Key k = (Eq k, Hashable k)
  218 
  219 -- | A 'NE.NonEmpty' list of 'HashMap's.
  220 --
  221 --   @since 0.1.0
  222 type Maps k v = NE.NonEmpty (HashMap k v)
