   31 --   which provides a canonical lens into that field.
   32 --
   33 --   @since 0.1.0
   34 module Language.Ninja.Misc.Annotated
   35   ( Annotated (..), annotation
   36   ) where
   37 
   38 import qualified Control.Lens as Lens
   39 
   40 --------------------------------------------------------------------------------
   41 
   42 -- | If you have some type that represents an AST node, it is often useful to
   43 --   add a polymorphic "annotation field" to it, which is used for things like
   44 --   source positions.
   45 --
   46 --   Specifically, suppose we have the following AST node type:
   47 --
   48 --   @data Foo = Foo { _fooBar :: !Bar, _fooBaz :: !Baz } deriving (…)@
   49 --
   50 --   Then an annotation field is added by the following process:
   51 --
   52 --   1. Add an extra (final) type parameter @ann@ to the type.
   53 --   2. Add an extra field @_fooAnn :: !ann@.
   54 --   3. Derive instances of 'Functor', 'Foldable', and 'Traversable'.
   55 --   4. If the type is recursive, add a 'Lens.Plated' instance.
   56 --      See "Language.Ninja.AST.Expr" for a complete example of this.
   57 --   5. Write an 'Annotated' instance with the canonical lens given by the
   58 --      @_fooAnn@ field. There are plenty of examples around this library.
   59 --
   60 --   The end result then looks like:
   61 --
   62 --   > data Foo ann
   63 --   >   = Foo
   64 --   >     { _fooAnn :: !ann
   65 --   >     , _fooBar :: !Bar
   66 --   >     , _fooBaz :: !Baz
   67 --   >     }
   68 --   >   deriving (…, Functor, Foldable, Traversable)
   69 --   >
   70 --   > instance Annotated Foo where
   71 --   >   annotation' = …
   72 --
   73 --   @since 0.1.0
   74 class (Functor ty) => Annotated (ty :: * -> *) where
   75   -- | Given a function that is used when 'fmap'ing any subterms, return a lens
   76   --   into the "annotation" field.
   77   --
   78   --   When writing an instance, keep in mind that @'annotation'' id@ should
   79   --   just be the typical definition for a lens into the annotation field.
   80   --
   81   --   prop> annotation' (f . g) == annotation' f . annotation' g
   82   --
   83   --   @since 0.1.0
   84   annotation' :: (ann -> ann') -> Lens.Lens (ty ann) (ty ann') ann ann'
   85 
   86 -- | This is just shorthand for @'annotation'' id@.
   87 --
   88 --   @since 0.1.0
   89 annotation :: (Annotated ty) => Lens.Lens' (ty ann) ann
   90 annotation = annotation' id
