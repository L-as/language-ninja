   77 -- | A set of Ninja build dependencies.
   78 --
   79 --   @since 0.1.0
   80 data Deps ann
   81   = MkDeps
   82     { _depsAnn       :: !ann
   83     , _depsNormal    :: !(HashSet Text)
   84     , _depsImplicit  :: !(HashSet Text)
   85     , _depsOrderOnly :: !(HashSet Text)
   86     }
   87   deriving (Eq, Show, Generic, Functor, Foldable, Traversable)
   88 
   89 -- | Construct a 'Deps' with all default values
   90 --
   91 --   @since 0.1.0
   92 {-# INLINE makeDeps #-}
   93 makeDeps :: (Monoid ann) => Deps ann
   94 makeDeps = MkDeps
   95             { _depsAnn       = mempty
   96             , _depsNormal    = mempty
   97             , _depsImplicit  = mempty
   98             , _depsOrderOnly = mempty
   99             }
  100 
  101 -- | A lens into the set of normal dependencies in a 'Deps'.
  102 --
  103 --   @since 0.1.0
  104 {-# INLINE depsNormal #-}
  105 depsNormal :: Lens.Lens' (Deps ann) (HashSet Text)
  106 depsNormal = Lens.lens _depsNormal
  107              $ \(MkDeps {..}) x -> MkDeps { _depsNormal = x, .. }
  108 
  109 -- | A lens into the set of implicit dependencies in a 'Deps'.
  110 --
  111 --   @since 0.1.0
  112 {-# INLINE depsImplicit #-}
  113 depsImplicit :: Lens.Lens' (Deps ann) (HashSet Text)
  114 depsImplicit = Lens.lens _depsImplicit
  115                $ \(MkDeps {..}) x -> MkDeps { _depsImplicit = x, .. }
  116 
  117 -- | A lens into the set of order-only dependencies in a 'Deps'.
  118 --
  119 --   @since 0.1.0
  120 {-# INLINE depsOrderOnly #-}
  121 depsOrderOnly :: Lens.Lens' (Deps ann) (HashSet Text)
  122 depsOrderOnly = Lens.lens _depsOrderOnly
  123                 $ \(MkDeps {..}) x -> MkDeps { _depsOrderOnly = x, .. }
  124 
  125 -- | The usual definition for 'Misc.Annotated'.
  126 --
  127 --   @since 0.1.0
  128 instance Misc.Annotated Deps where
  129   annotation' _ = Lens.lens _depsAnn
  130                   $ \(MkDeps {..}) x -> MkDeps { _depsAnn = x, .. }
  131 
  132 -- | Converts to @{ann: …, normal: …, implicit: …, order-only: …}@.
  133 --
  134 --   @since 0.1.0
  135 instance (Aeson.ToJSON ann) => Aeson.ToJSON (Deps ann) where
  136   toJSON (MkDeps {..})
  137     = [ "ann"        .= _depsAnn
  138       , "normal"     .= _depsNormal
  139       , "implicit"   .= _depsImplicit
  140       , "order-only" .= _depsOrderOnly
  141       ] |> Aeson.object
  142 
  143 -- | Inverse of the 'Aeson.ToJSON' instance.
  144 --
  145 --   @since 0.1.0
  146 instance (Aeson.FromJSON ann) => Aeson.FromJSON (Deps ann) where
  147   parseJSON = (Aeson.withObject "Deps" $ \o -> do
  148                   _depsAnn       <- (o .: "ann")        >>= pure
  149                   _depsNormal    <- (o .: "normal")     >>= pure
  150                   _depsImplicit  <- (o .: "implicit")   >>= pure
  151                   _depsOrderOnly <- (o .: "order-only") >>= pure
  152                   pure (MkDeps {..}))
  153 
  154 -- | Reasonable 'QC.Arbitrary' instance for 'Deps'.
  155 --
  156 --   @since 0.1.0
  157 instance (QC.Arbitrary ann) => QC.Arbitrary (Deps ann) where
  158   arbitrary = MkDeps
  159               <$> QC.arbitrary
  160               <*> QC.arbitrary
  161               <*> QC.arbitrary
  162               <*> QC.arbitrary

  186 -- | The set of constraints required for a given constraint to be automatically
  187 --   computed for a 'Ninja'.
  188 --
  189 --   @since 0.1.0
  190 type DepsConstraint (c :: * -> Constraint) (ann :: *)
  191   = ( c (HashSet Text)
  192     , c ann
  193     )
