   31 --   A typeclass exposing an interface that allows reading a file from a
   32 --   filesystem (virtual or otherwise).
   33 --
   34 --   @since 0.1.0
   35 module Language.Ninja.Mock.ReadFile
   36   ( MonadReadFile (..)
   37   ) where
   38 
   39 import           Prelude                   (IO, Monad)
   40 
   41 import           Control.Monad.Trans.Class (MonadTrans (lift))
   42 
   43 import qualified Control.Lens              as Lens
   44 
   45 import           Data.Text                 (Text)
   46 import qualified Data.Text.IO              as Text
   47 
   48 import           Language.Ninja.Misc.Path  (Path, pathString)
   49 
   50 import           Flow                      ((.>))
   51 
   52 --------------------------------------------------------------------------------
   53 
   54 -- Remember: we have imported Prelude hiding 'readFile'.
   55 
   56 -- | This typeclass allows you to write code that reads files from the
   57 --   filesystem and then later run that code purely against a virtual filesystem
   58 --   of some description.
   59 --
   60 --   @since 0.1.0
   61 class (Monad m) => MonadReadFile m where
   62   {-# MINIMAL readFile #-}
   63 
   64   -- | Read the file located at the given path and decode it into 'Text'.
   65   --
   66   --   FIXME: some notion of error handling should be encoded into the type
   67   --
   68   --   @since 0.1.0
   69   readFile :: Path -> m Text
   70 
   71 -- | The obvious instance for 'IO'.
   72 --
   73 --   @since 0.1.0
   74 instance MonadReadFile IO where
   75   readFile = Lens.view pathString .> Text.readFile
   76 
   77 -- | A placeholder (undecidable) instance that allows this constraint to
   78 --   propagate through monad transformers without needing to 'lift' manually.
   79 --
   80 --   @since 0.1.0
   81 instance ( MonadTrans t, Monad (t m), MonadReadFile m
   82          ) => MonadReadFile (t m) where
   83   readFile = readFile .> lift
