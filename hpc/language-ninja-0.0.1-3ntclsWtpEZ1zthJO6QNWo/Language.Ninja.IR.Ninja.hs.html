   80 -- | A parsed and normalized Ninja file.
   81 --
   82 --   @since 0.1.0
   83 data Ninja
   84   = MkNinja
   85     { _ninjaMeta     :: !Meta
   86     , _ninjaBuilds   :: !(HashSet Build)
   87     , _ninjaPhonys   :: !(HashMap Target (HashSet Target))
   88     , _ninjaDefaults :: !(HashSet Target)
   89     , _ninjaPools    :: !(HashSet Pool)
   90     }
   91   deriving (Eq, Show, Generic)
   92 
   93 -- | Construct a default 'Ninja' value.
   94 --
   95 --   @since 0.1.0
   96 {-# INLINE makeNinja #-}
   97 makeNinja :: Ninja
   98 makeNinja = MkNinja
   99             { _ninjaMeta     = Ninja.makeMeta
  100             , _ninjaBuilds   = HS.empty
  101             , _ninjaPhonys   = HM.empty
  102             , _ninjaDefaults = HS.empty
  103             , _ninjaPools    = HS.empty
  104             }
  105 
  106 -- | Metadata, which includes top-level variables like @builddir@.
  107 --
  108 --   @since 0.1.0
  109 {-# INLINE ninjaMeta #-}
  110 ninjaMeta :: Lens.Lens' Ninja Meta
  111 ninjaMeta = Lens.lens _ninjaMeta
  112             $ \(MkNinja {..}) x -> MkNinja { _ninjaMeta = x, .. }
  113 
  114 -- | Compiled @build@ declarations.
  115 --
  116 --   @since 0.1.0
  117 {-# INLINE ninjaBuilds #-}
  118 ninjaBuilds :: Lens.Lens' Ninja (HashSet Build)
  119 ninjaBuilds = Lens.lens _ninjaBuilds
  120               $ \(MkNinja {..}) x -> MkNinja { _ninjaBuilds = x, .. }
  121 
  122 -- | Phony targets, as documented
  123 --   <https://ninja-build.org/manual.html#_more_details here>.
  124 --
  125 --   @since 0.1.0
  126 {-# INLINE ninjaPhonys #-}
  127 ninjaPhonys :: Lens.Lens' Ninja (HashMap Target (HashSet Target))
  128 ninjaPhonys = Lens.lens _ninjaPhonys
  129               $ \(MkNinja {..}) x -> MkNinja { _ninjaPhonys = x, .. }
  130 
  131 -- | The set of default targets, as documented
  132 --   <https://ninja-build.org/manual.html#_default_target_statements here>.
  133 --
  134 --   @since 0.1.0
  135 {-# INLINE ninjaDefaults #-}
  136 ninjaDefaults :: Lens.Lens' Ninja (HashSet Target)
  137 ninjaDefaults = Lens.lens _ninjaDefaults
  138                 $ \(MkNinja {..}) x -> MkNinja { _ninjaDefaults = x, .. }
  139 
  140 -- | The set of pools for this Ninja file.
  141 --
  142 --   @since 0.1.0
  143 {-# INLINE ninjaPools #-}
  144 ninjaPools :: Lens.Lens' Ninja (HashSet Pool)
  145 ninjaPools = Lens.lens _ninjaPools
  146              $ \(MkNinja {..}) x -> MkNinja { _ninjaPools = x, .. }
  147 
  148 -- | Converts to @{meta: …, builds: …, phonys: …, defaults: …, pools: …}@.
  149 --
  150 --   @since 0.1.0
  151 instance Aeson.ToJSON Ninja where
  152   toJSON (MkNinja {..})
  153     = [ "meta"     .= _ninjaMeta
  154       , "builds"   .= _ninjaBuilds
  155       , "phonys"   .= _ninjaPhonys
  156       , "defaults" .= _ninjaDefaults
  157       , "pools"    .= _ninjaPools
  158       ] |> Aeson.object
  159 
  160 -- | Inverse of the 'Aeson.ToJSON' instance.
  161 --
  162 --   @since 0.1.0
  163 instance Aeson.FromJSON Ninja where
  164   parseJSON = (Aeson.withObject "Ninja" $ \o -> do
  165                   _ninjaMeta     <- (o .: "meta")     >>= pure
  166                   _ninjaBuilds   <- (o .: "builds")   >>= pure
  167                   _ninjaPhonys   <- (o .: "phonys")   >>= pure
  168                   _ninjaDefaults <- (o .: "defaults") >>= pure
  169                   _ninjaPools    <- (o .: "pools")    >>= pure
  170                   pure (MkNinja {..}))
  171

  192 -- | The set of constraints required for a given constraint to be automatically
  193 --   computed for a 'Ninja'.
  194 --
  195 --   @since 0.1.0
  196 type NinjaConstraint (c :: * -> Constraint)
  197   = ( c Text
  198     , c Ver.Version
  199     , c (HashMap Target (HashSet Target))
  200     , c (HashSet Build)
  201     , c (HashSet Target)
  202     , c (HashSet Pool)
  203     )
