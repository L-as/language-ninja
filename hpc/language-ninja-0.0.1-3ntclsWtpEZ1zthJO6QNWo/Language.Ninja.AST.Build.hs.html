   78 -- | A parsed Ninja @build@ declaration.
   79 --
   80 --   @since 0.1.0
   81 data Build ann
   82   = MkBuild
   83     { _buildAnn  :: !ann
   84     , _buildRule :: !Text
   85     , _buildEnv  :: !(AST.Env Text Text)
   86     , _buildDeps :: !(AST.Deps ann)
   87     , _buildBind :: !(HashMap Text Text)
   88     }
   89   deriving (Eq, Show, Generic, Functor, Foldable, Traversable)
   90 
   91 -- | Construct a 'Build' with all default values.
   92 --
   93 --   @since 0.1.0
   94 {-# INLINE makeBuild #-}
   95 makeBuild :: (Monoid ann)
   96           => Text
   97           -- ^ The rule name
   98           -> AST.Env Text Text
   99           -- ^ The environment
  100           -> Build ann
  101 makeBuild rule env = MkBuild
  102                      { _buildAnn  = mempty
  103                      , _buildRule = rule
  104                      , _buildEnv  = env
  105                      , _buildDeps = AST.makeDeps
  106                      , _buildBind = mempty
  107                      }
  108 
  109 -- | A lens into the rule name associated with a 'Build'.
  110 --
  111 --   @since 0.1.0
  112 {-# INLINE buildRule #-}
  113 buildRule :: Lens.Lens' (Build ann) Text
  114 buildRule = Lens.lens _buildRule
  115             $ \(MkBuild {..}) x -> MkBuild { _buildRule = x, .. }
  116 
  117 -- | A lens into the environment associated with a 'Build'.
  118 --
  119 --   @since 0.1.0
  120 {-# INLINE buildEnv #-}
  121 buildEnv :: Lens.Lens' (Build ann) (AST.Env Text Text)
  122 buildEnv = Lens.lens _buildEnv
  123            $ \(MkBuild {..}) x -> MkBuild { _buildEnv = x, .. }
  124 
  125 -- | A lens into the dependencies associated with a 'Build'.
  126 --
  127 --   @since 0.1.0
  128 {-# INLINE buildDeps #-}
  129 buildDeps :: Lens.Lens' (Build ann) (AST.Deps ann)
  130 buildDeps = Lens.lens _buildDeps
  131             $ \(MkBuild {..}) x -> MkBuild { _buildDeps = x, .. }
  132 
  133 -- | A lens into the bindings associated with a 'Build'.
  134 --
  135 --   @since 0.1.0
  136 {-# INLINE buildBind #-}
  137 buildBind :: Lens.Lens' (Build ann) (HashMap Text Text)
  138 buildBind = Lens.lens _buildBind
  139             $ \(MkBuild {..}) x -> MkBuild { _buildBind = x, .. }
  140 
  141 -- | The usual definition for 'Misc.Annotated'.
  142 --
  143 --   @since 0.1.0
  144 instance Misc.Annotated Build where
  145   annotation' f = Lens.lens (helper .> fst) (helper .> snd)
  146     where
  147       helper (MkBuild {..})
  148         = ( _buildAnn
  149           , \x -> MkBuild { _buildAnn = x, _buildDeps = f <$> _buildDeps, .. } )
  150 
  151 -- | Converts to @{ann: …, rule: …, env: …, deps: …, bind: …}@.
  152 --
  153 --   @since 0.1.0
  154 instance (Aeson.ToJSON ann) => Aeson.ToJSON (Build ann) where
  155   toJSON (MkBuild {..})
  156     = [ "ann"  .= _buildAnn
  157       , "rule" .= _buildRule
  158       , "env"  .= _buildEnv
  159       , "deps" .= _buildDeps
  160       , "bind" .= _buildBind
  161       ] |> Aeson.object
  162 
  163 -- | Inverse of the 'Aeson.ToJSON' instance.
  164 --
  165 --   @since 0.1.0
  166 instance (Aeson.FromJSON ann) => Aeson.FromJSON (Build ann) where
  167   parseJSON = (Aeson.withObject "Build" $ \o -> do
  168                   _buildAnn  <- (o .: "ann")  >>= pure
  169                   _buildRule <- (o .: "rule") >>= pure
  170                   _buildEnv  <- (o .: "env")  >>= pure
  171                   _buildDeps <- (o .: "deps") >>= pure
  172                   _buildBind <- (o .: "bind") >>= pure
  173                   pure (MkBuild {..}))

  177 --   @since 0.1.0
  178 instance (QC.Arbitrary ann) => QC.Arbitrary (Build ann) where
  179   arbitrary = MkBuild

  208 -- | The set of constraints required for a given constraint to be automatically
  209 --   computed for a 'Build'.
  210 --
  211 --   @since 0.1.0
  212 type BuildConstraint (c :: * -> Constraint) (ann :: *)
  213   = ( AST.DepsConstraint c ann
  214     , c Text
  215     , c (HashMap Text Text)
  216     , c (AST.Maps Text Text)
  217     , c ann
  218     )
