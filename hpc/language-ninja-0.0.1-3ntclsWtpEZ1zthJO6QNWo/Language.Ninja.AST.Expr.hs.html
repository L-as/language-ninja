   89 -- | An expression containing variable references in the Ninja language.
   90 --
   91 --   @since 0.1.0
   92 data Expr ann
   93   = -- | Sequencing of expressions.
   94     --
   95     --   @since 0.1.0
   96     Exprs !ann ![Expr ann]
   97   | -- | A literal string.
   98     --
   99     --   @since 0.1.0
  100     Lit   !ann !Text
  101   | -- | A variable reference.
  102     --
  103     --   @since 0.1.0
  104     Var   !ann !Text
  105   deriving (Eq, Show, Generic, Data, Functor, Foldable, Traversable)
  106 
  107 -- | A prism for the 'Exprs' constructor.
  108 --
  109 --   @since 0.1.0
  110 {-# INLINE _Exprs #-}
  111 _Exprs :: Lens.Prism' (Expr ann) (ann, [Expr ann])
  112 _Exprs = Lens.prism' (uncurry Exprs)
  113          $ \case (Exprs ann es) -> Just (ann, es)
  114                  _              -> Nothing
  115 
  116 -- | A prism for the 'Lit' constructor.
  117 --
  118 --   @since 0.1.0
  119 {-# INLINE _Lit #-}
  120 _Lit :: Lens.Prism' (Expr ann) (ann, Text)
  121 _Lit = Lens.prism' (uncurry Lit)
  122        $ \case (Lit ann text) -> Just (ann, text)
  123                _              -> Nothing
  124 
  125 -- | A prism for the 'Var' constructor.
  126 --
  127 --   @since 0.1.0
  128 {-# INLINE _Var #-}
  129 _Var :: Lens.Prism' (Expr ann) (ann, Text)
  130 _Var = Lens.prism' (uncurry Var)
  131        $ \case (Var ann name) -> Just (ann, name)
  132                _              -> Nothing
  133 
  134 -- | Evaluate the given 'Expr' in the given context (@'Env' 'Text' 'Text'@).
  135 --
  136 --   @since 0.1.0
  137 askExpr :: AST.Env Text Text -> Expr ann -> Text
  138 askExpr e (Exprs _ xs) = Text.concat (map (askExpr e) xs)
  139 askExpr _ (Lit   _ x)  = x
  140 askExpr e (Var   _ x)  = askVar e x
  141 
  142 -- | Look up the given variable in the given context, returning the empty string
  143 --   if the variable was not found.
  144 --
  145 --   @since 0.1.0
  146 askVar :: AST.Env Text Text -> Text -> Text
  147 askVar e x = fromMaybe Text.empty (AST.askEnv e x)
  148 
  149 -- | Add a binding with the given name ('Text') and value ('Expr') to the
  150 --   given context.
  151 --
  152 --   @since 0.1.0
  153 addBind :: Text -> Expr ann -> AST.Env Text Text -> AST.Env Text Text
  154 addBind k v e = AST.addEnv k (askExpr e v) e
  155 
  156 -- | Add bindings from a list. Note that this function evaluates all the
  157 --   right-hand-sides first, and then adds them all to the environment.
  158 --
  159 --   For example:
  160 --
  161 --   >>> let binds = [("x", Lit "5"), ("y", Var "x")]
  162 --   >>> AST.headEnv (addBinds binds AST.makeEnv)
  163 --   fromList [("x","5"),("y","")]
  164 --
  165 --   @since 0.1.0
  166 addBinds :: [(Text, Expr ann)] -> AST.Env Text Text -> AST.Env Text Text
  167 addBinds bs e = map (second (askExpr e) .> uncurry AST.addEnv .> Endo) bs
  168                 |> mconcat
  169                 |> (\endo -> appEndo endo e)
  170 
  171 -- | Normalize an 'Expr' by recursively flattening any 'Exprs' nodes, removing
  172 --   empty 'Lit' nodes, combining adjacent 'Lit' nodes, and pulling out the
  173 --   interior of the top-level 'Exprs' node if it has only one subnode.
  174 --
  175 --   The number of 'Exprs' nodes in the output is guaranteed to be 0 or 1.
  176 --
  177 --   If it is 0, then there is exactly one node of any type in the output.
  178 --
  179 --   The output is thus isomorphic to @(Maybe ann, [(ann, Either Text Text)])@,
  180 --   where the @Maybe ann@ represents the annotation of the top-level 'Exprs'
  181 --   node if it exists.
  182 --
  183 --   @since 0.1.0
  184 normalizeExpr :: forall ann. (Monoid ann) => Expr ann -> Expr ann
  185 normalizeExpr = flatten .> removeEmpty .> combineAdj .> listToExpr
  186   where
  187     flatten :: Expr ann -> [Expr ann]
  188     flatten (Exprs _ es) = concatMap flatten es
  189     flatten owise        = [owise]
  190 
  191     removeEmpty :: [Expr ann] -> [Expr ann]
  192     removeEmpty []                = []
  193     removeEmpty (Lit _ "" : rest) = removeEmpty rest
  194     removeEmpty (owise    : rest) = owise : removeEmpty rest
  195 
  196     combineAdj :: [Expr ann] -> [Expr ann]
  197     combineAdj = (\case
  198       []                               -> []
  199       (Lit annX x : Lit annY y : rest) -> (Lit (annX <> annY) (x <> y))
  200                                           |> (\e -> combineAdj (e : rest))
  201       (owise                   : rest) -> owise : combineAdj rest)
  202 
  203     listToExpr :: [Expr ann] -> Expr ann
  204     listToExpr [e] = e
  205     listToExpr es  = Exprs (mconcat (map (Lens.view Misc.annotation) es)) es

  212 -- | The usual definition for 'Misc.Annotated'.
  213 --
  214 --   @since 0.1.0
  215 instance Misc.Annotated Expr where
  216   annotation' f = Lens.lens (helper .> fst) (helper .> snd)
  217     where
  218       helper (Exprs ann   es) = (ann, \x -> Exprs x (map (fmap f) es))
  219       helper (Lit   ann text) = (ann, \x -> Lit   x text)
  220       helper (Var   ann name) = (ann, \x -> Var   x name)
  221 
  222 -- | Converts 'Exprs' to @{ann: …, exprs: […]}@, 'Lit' to @{ann: …, lit: […]}@,
  223 --   and 'Var' to @{ann: …, var: …}@.
  224 --
  225 --   @since 0.1.0
  226 instance (Aeson.ToJSON ann) => Aeson.ToJSON (Expr ann) where
  227   toJSON (Exprs ann   es) = Aeson.object ["ann" .= ann, "exprs" .= es]
  228   toJSON (Lit   ann text) = Aeson.object ["ann" .= ann, "lit"   .= text]
  229   toJSON (Var   ann name) = Aeson.object ["ann" .= ann, "var"   .= name]
  230 
  231 -- | Inverse of the 'Aeson.ToJSON' instance.
  232 --
  233 --   @since 0.1.0
  234 instance (Aeson.FromJSON ann) => Aeson.FromJSON (Expr ann) where
  235   parseJSON = Aeson.withObject "Expr" $ \o -> do
  236     ann <- o .: "ann"
  237     asum [ Exprs ann <$> (o .: "exprs")
  238          , Lit   ann <$> (o .: "lit")
  239          , Var   ann <$> (o .: "var")
  240          ]

  242 -- | Reasonable 'QC.Arbitrary' instance for 'Expr'.
  243 --
  244 --   @since 0.1.0
  245 instance forall ann. (QC.Arbitrary ann) => QC.Arbitrary (Expr ann) where
  246   arbitrary = QC.sized go
  247     where
  248       go :: Int -> QC.Gen (Expr ann)
  249       go n | n <= 0 = [ litG (QC.resize litLength QC.arbitrary)
  250                       , varG (QC.resize varLength QC.arbitrary)
  251                       ] |> QC.oneof
  252       go n          = [ go 0
  253                       , do width <- (`mod` maxWidth) <$> QC.arbitrary
  254                            let subtree = go (n `div` lossRate)
  255                            Exprs <$> QC.arbitrary <*> QC.vectorOf width subtree
  256                       ] |> QC.oneof
  257 
  258       litG, varG :: QC.Gen Text -> QC.Gen (Expr ann)
  259       litG g = Lit <$> QC.arbitrary <*> g
  260       varG g = Var <$> QC.arbitrary <*> g
  261 
  262       litLength, varLength, lossRate, maxWidth :: Int
  263       litLength = 10
  264       varLength = 10
  265       maxWidth  = 5
  266       lossRate  = 2
