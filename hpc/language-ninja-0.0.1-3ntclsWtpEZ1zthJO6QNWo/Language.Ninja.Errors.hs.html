  108 -- | This type subsumes any error that can be thrown during execution of a
  109 --   function defined in @language-ninja@.
  110 --
  111 --   @since 0.1.0
  112 data NinjaError
  113   = -- | Generic catch-all error constructor. Avoid using this.
  114     --
  115     --   @since 0.1.0
  116     GenericNinjaError !Text
  117   | -- | Errors encountered during parsing.
  118     --
  119     --   @since 0.1.0
  120     NinjaParseError   !ParseError
  121   | -- | Errors encountered during compilation.
  122     --
  123     --   @since 0.1.0
  124     NinjaCompileError !CompileError
  125   deriving (Eq, Show, Generic)
  126 
  127 instance Exception NinjaError
  128 
  129 -- | Throw a 'NinjaError'.
  130 --
  131 --   @since 0.1.0
  132 throwNinjaError :: (MonadError NinjaError m) => NinjaError -> m a
  133 throwNinjaError = throwError
  134 
  135 -- | Throw a generic catch-all 'NinjaError'.
  136 --
  137 --   @since 0.1.0
  138 throwGenericNinjaError :: (MonadError NinjaError m) => Text -> m a
  139 throwGenericNinjaError msg = throwNinjaError (GenericNinjaError msg)
  140 
  141 -- | Throw a 'ParseError'.
  142 --
  143 --   @since 0.1.0
  144 throwNinjaParseError :: (MonadError NinjaError m) => ParseError -> m a
  145 throwNinjaParseError e = throwNinjaError (NinjaParseError e)
  146 
  147 -- | Throw a 'CompileError'.
  148 --
  149 --   @since 0.1.0
  150 throwNinjaCompileError :: (MonadError NinjaError m) => CompileError -> m a
  151 throwNinjaCompileError e = throwNinjaError (NinjaCompileError e)
