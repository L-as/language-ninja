   64 -- | This type represents a Unix path string.
   65 --
   66 --   @since 0.1.0
   67 newtype Path
   68   = MkPath
   69     { _pathIText :: IText
   70     }
   71   deriving ( Eq, Ord, Show, Read, IsString, Generic, Hashable, NFData
   72            , Aeson.ToJSON, Aeson.FromJSON, Aeson.ToJSONKey, Aeson.FromJSONKey )
   73 
   74 -- | Construct a 'Path' from some 'Text'.
   75 --
   76 --   @since 0.1.0
   77 {-# INLINE makePath #-}
   78 makePath :: Text -> Path
   79 makePath = Lens.view Ninja.itext .> MkPath
   80 
   81 -- | An isomorphism between a 'Path' and its underlying 'IText'.
   82 --
   83 --   @since 0.1.0
   84 {-# INLINE pathIText #-}
   85 pathIText :: Lens.Iso' Path IText
   86 pathIText = Lens.iso _pathIText MkPath
   87 
   88 -- | An isomorphism that gives access to a 'Text'-typed view of a 'Path',
   89 --   even though the underlying data has type 'IText'.
   90 --
   91 --   This is equivalent to @pathIText . from Ninja.itext@.
   92 --
   93 --   @since 0.1.0
   94 {-# INLINE pathText #-}
   95 pathText :: Lens.Iso' Path Text
   96 pathText = pathIText . Lens.from Ninja.itext
   97 
   98 -- | An isomorphism that gives access to a 'String'-typed view of a 'Path'.
   99 --
  100 --   @since 0.1.0
  101 {-# INLINE pathString #-}
  102 pathString :: Lens.Iso' Path String
  103 pathString = pathText . Lens.iso Text.unpack Text.pack
  104 
  105 -- | An isomorphism between a 'Path' and a 'FP.FilePath' from @system-filepath@.
  106 --   This uses 'FP.decodeString' and 'FP.encodeString', so all the caveats on
  107 --   those functions apply here.
  108 --
  109 --   @since 0.1.0
  110 {-# INLINE pathFP #-}
  111 pathFP :: Lens.Iso' Path FP.FilePath
  112 pathFP = pathString . Lens.iso FP.decodeString FP.encodeString
  113 
  114 -- | Uses the underlying 'IText' instance.
  115 --
  116 --   @since 0.1.0
  117 instance (Monad m, SC.Serial m Text) => SC.Serial m Path where
  118   series = SC.newtypeCons MkPath
  119 
  120 -- | Uses the underlying 'IText' instance.
  121 --
  122 --   @since 0.1.0
  123 instance (Monad m, SC.CoSerial m Text) => SC.CoSerial m Path where
  124   coseries rs = SC.newtypeAlts rs
  125                 >>- \f -> pure (_pathIText .> f)
