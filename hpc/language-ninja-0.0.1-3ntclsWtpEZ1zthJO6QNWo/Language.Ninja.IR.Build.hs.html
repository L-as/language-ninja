   70 -- | A Ninja @build@ declaration, as documented
   71 --   <https://ninja-build.org/manual.html#_build_statements here>.
   72 --
   73 --   @since 0.1.0
   74 data Build
   75   = MkBuild
   76     { _buildRule :: !Rule
   77     , _buildOuts :: !(HashSet Output)
   78     , _buildDeps :: !(HashSet Dependency)
   79     }
   80   deriving (Eq, Show, Generic)
   81 
   82 -- | Construct a default 'Build' from the given 'Rule'
   83 --
   84 --   @since 0.1.0
   85 {-# INLINE makeBuild #-}
   86 makeBuild :: Rule -> Build
   87 makeBuild rule = MkBuild
   88                  { _buildRule = rule
   89                  , _buildOuts = HS.empty
   90                  , _buildDeps = HS.empty
   91                  }
   92 
   93 -- | The rule to execute when building any of the outputs.
   94 --
   95 --   @since 0.1.0
   96 {-# INLINE buildRule #-}
   97 buildRule :: Lens.Lens' Build Rule
   98 buildRule = Lens.lens _buildRule
   99             $ \(MkBuild {..}) x -> MkBuild { _buildRule = x, .. }
  100 
  101 -- | The outputs that are built as a result of rule execution.
  102 --
  103 --   @since 0.1.0
  104 {-# INLINE buildOuts #-}
  105 buildOuts :: Lens.Lens' Build (HashSet Output)
  106 buildOuts = Lens.lens _buildOuts
  107             $ \(MkBuild {..}) x -> MkBuild { _buildOuts = x, .. }
  108 
  109 -- | The dependencies that must be satisfied before this can be built.
  110 --
  111 --   @since 0.1.0
  112 {-# INLINE buildDeps #-}
  113 buildDeps :: Lens.Lens' Build (HashSet Dependency)
  114 buildDeps = Lens.lens _buildDeps
  115             $ \(MkBuild {..}) x -> MkBuild { _buildDeps = x, .. }
  116 
  117 -- | Converts to @{rule: …, outputs: …, dependencies: …}@.
  118 --
  119 --   @since 0.1.0
  120 instance Aeson.ToJSON Build where
  121   toJSON (MkBuild {..})
  122     = [ "rule"         .= _buildRule
  123       , "outputs"      .= _buildOuts
  124       , "dependencies" .= _buildDeps
  125       ] |> Aeson.object
  126 
  127 -- | Inverse of the 'Aeson.ToJSON' instance.
  128 --
  129 --   @since 0.1.0
  130 instance Aeson.FromJSON Build where
  131   parseJSON = (Aeson.withObject "Build" $ \o -> do
  132                   _buildRule <- (o .: "rule")         >>= pure
  133                   _buildOuts <- (o .: "outputs")      >>= pure
  134                   _buildDeps <- (o .: "dependencies") >>= pure
  135                   pure (MkBuild {..}))
  136 
  137 -- | Default 'Hashable' instance via 'Generic'.
  138 --
  139 --   @since 0.1.0
  140 instance Hashable Build
  141 
  142 -- | Default 'NFData' instance via 'Generic'.
  143 --
  144 --   @since 0.1.0
  145 instance NFData Build
  146 
  147 -- | Default 'SC.Serial' instance via 'Generic'.
  148 --
  149 --   @since 0.1.0
  150 instance (Monad m, BuildConstraint (SC.Serial m)) => SC.Serial m Build
  151 
  152 -- | Default 'SC.CoSerial' instance via 'Generic'.
  153 --
  154 --   @since 0.1.0
  155 instance (Monad m, BuildConstraint (SC.CoSerial m)) => SC.CoSerial m Build
  156 
  157 -- | The set of constraints required for a given constraint to be automatically
  158 --   computed for a 'Build'.
  159 --
  160 --   @since 0.1.0
  161 type BuildConstraint (c :: * -> Constraint)
  162   = ( c Text
  163     , c (HashSet Output)
  164     , c (HashSet Dependency)
  165     )
